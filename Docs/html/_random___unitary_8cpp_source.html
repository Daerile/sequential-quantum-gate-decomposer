34 void few_CNOT_unitary( int qbit_num, int cnot_num, QGD_Complex16* mtx) {
35 
36  // the current number of CNOT gates
37  int cnot_num_curr = 0;
38 
39  // the size of the matrix
40  int matrix_size = Power_of_2(qbit_num);
41 
42  // The unitary discribing each qubits in their initial state
43  memset( mtx, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );
44  create_identity( mtx, matrix_size );
45 
46  QGD_Complex16* mtx_tmp = (QGD_Complex16*)qgd_calloc( matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
47 
48  QGD_Complex16* gate_matrix = (QGD_Complex16*)qgd_calloc( matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
49  memset( gate_matrix, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );
50 
51  // constructing the unitary
52  while (true) {
53  int cnot_or_u3 = rand() % 5 + 1;
54 
55  CNOT* cnot_op = NULL;
56  U3* u3_op = NULL;
57 
58  if (cnot_or_u3 <= 4) {
59  // creating random parameters for the U3 operation
60  double parameters[3];
61 
62  parameters[0] = double(rand())/RAND_MAX*4*M_PI;
63  parameters[1] = double(rand())/RAND_MAX*2*M_PI;
64  parameters[2] = double(rand())/RAND_MAX*2*M_PI;
65 
66 
67  // randomly choose the target qbit
68  int target_qbit = rand() % qbit_num;
69 
70  // creating the U3 gate
71  u3_op = new U3(qbit_num, target_qbit, true, true, true);
72 
73  // get the matrix of the operation
74  u3_op->matrix(parameters, gate_matrix);
75  }
76  else if ( cnot_or_u3 == 5 ) {
77  // randomly choose the target qbit
78  int target_qbit = rand() % qbit_num;
79 
80  // randomly choose the control qbit
81  int control_qbit = rand() % qbit_num;
82 
83  if (target_qbit == control_qbit) {
84  create_identity( gate_matrix, matrix_size );
85  }
86  else {
87 
88  // creating the CNOT gate
89  cnot_op = new CNOT(qbit_num, control_qbit, target_qbit);
90 
91  // get the matrix of the operation
92  cnot_op->matrix(gate_matrix);
93 
94  cnot_num_curr = cnot_num_curr + 1;
95  }
96  }
97  else {
98  create_identity( gate_matrix, matrix_size );
99  }
100 
101 
102  // get the current unitary
103  zgemm3m_wrapper(gate_matrix, mtx, mtx_tmp, matrix_size);
104  memset( gate_matrix, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );
105  memcpy( mtx, mtx_tmp, matrix_size*matrix_size*sizeof(QGD_Complex16) );
106 
107 
108  delete u3_op;
109  u3_op = NULL;
110 
111  delete cnot_op;
112  cnot_op = NULL;
113 
114 
115 
116 
117  // exit the loop if the maximal number of CNOT gates reched
118  if (cnot_num_curr >= cnot_num) {
119 
120  if (mtx_tmp != NULL) {
121  qgd_free( mtx_tmp );
122  mtx_tmp = NULL;
123  }
124 
125  if (gate_matrix != NULL) {
126  qgd_free( gate_matrix );
127  gate_matrix = NULL;
128  }
129 
130  return;
131  }
132 
133  }
