130 static int
131 qgd_N_Qubit_Decomposition_Wrapper_init(qgd_N_Qubit_Decomposition_Wrapper *self, PyObject *args, PyObject *kwds)
132 {
133  // The tuple of expected keywords
134  static char *kwlist[] = {(char*)"Umtx", (char*)"qbit_num", (char*)"optimize_layer_num", (char*)"initial_guess", NULL};
135 
136  // initiate variables for input arguments
137  PyObject *Umtx_arg = NULL;
138  int qbit_num = -1;
139  bool optimize_layer_num = false;
140  PyObject *initial_guess = NULL;
141 
142  // parsing input arguments
143  if (!PyArg_ParseTupleAndKeywords(args, kwds, "|OibO", kwlist,
144  &Umtx_arg, &qbit_num, &optimize_layer_num, &initial_guess))
145  return -1;
146 
147  // convert python object array to numpy C API array
148  if ( Umtx_arg == NULL ) return -1;
149  self->Umtx = PyArray_FROM_OTF(Umtx_arg, NPY_COMPLEX128, NPY_ARRAY_IN_ARRAY);
150 
151  // test C-style contiguous memory allocation of the array
152  if ( !PyArray_IS_C_CONTIGUOUS(self->Umtx) ) {
153  std::cout << "Umtx is not memory contiguous" << std::endl;
154  }
155 
156  // get the dimensions of the array self->Umtx
157  int dim_num = PyArray_NDIM( self->Umtx );
158  npy_intp* dims = PyArray_DIMS(self->Umtx);
159 
160  // insert a test for dimensions
161  if (dim_num != 2) {
162  std::cout << "The number of dimensions of the input matrix should be 2, but Umtx with " << dim_num << " dimensions was given" << std::endl;
163  return -1;
164  }
165 
166  // get the pointer to the data stored in the matrix self->Umtx
167  QGD_Complex16* data = (QGD_Complex16*)PyArray_DATA(self->Umtx);
168 
169 
170  // create QGD version of the Umtx
171  Matrix Umtx_mtx = Matrix(data, dims[0], dims[1]);
172 
173 
174  // determine the initial guess type
175  PyObject* initial_guess_string = PyObject_Str(initial_guess);
176  PyObject* initial_guess_string_unicode = PyUnicode_AsEncodedString(initial_guess_string, "utf-8", "~E~");
177  const char* initial_guess_C = PyBytes_AS_STRING(initial_guess_string_unicode);
178 
179  guess_type qgd_initial_guess;
180  if ( strcmp("zeros", initial_guess_C) == 0 or strcmp("ZEROS", initial_guess_C) == 0) {
181  qgd_initial_guess = ZEROS;
182  }
183  else if ( strcmp("random", initial_guess_C)==0 or strcmp("RANDOM", initial_guess_C)==0) {
184  qgd_initial_guess = RANDOM;
185  }
186  else if ( strcmp("close_to_zero", initial_guess_C)==0 or strcmp("CLOSE_TO_ZERO", initial_guess_C)==0) {
187  qgd_initial_guess = CLOSE_TO_ZERO;
188  }
189  else {
190  std::cout << "Wring initial guess format. Using default ZEROS." << std::endl;
191  qgd_initial_guess = ZEROS;
192  }
193 
194  // create an instance of the class N_Qubit_Decomposition
195  if (qbit_num > 0 ) {
196  self->decomp = create_N_Qubit_Decomposition( Umtx_mtx, qbit_num, optimize_layer_num, qgd_initial_guess);
197  }
198  else {
199  std::cout << "The number of qubits should be given as a positive integer, " << qbit_num << " was given" << std::endl;
200  return -1;
201  }
202 
203 
204 
205  Py_XDECREF(initial_guess_string);
206  Py_XDECREF(initial_guess_string_unicode);
207 
208  return 0;
209 }

270 static PyObject *
271 get_operation( N_Qubit_Decomposition<Sub_Matrix_Decomposition>* decomp, int &idx ) {
272 
273 
274  // create dictionary conatining the gate data
275  PyObject* gate = PyDict_New();
276 
277  Operation* operation = decomp->get_operation( idx );
278 
279  if (operation->get_type() == CNOT_OPERATION) {
280 
281  // create gate parameters
282  PyObject* type = Py_BuildValue("s", "CNOT" );
283  PyObject* target_qbit = Py_BuildValue("i", operation->get_target_qbit() );
284  PyObject* control_qbit = Py_BuildValue("i", operation->get_control_qbit() );
285 
286 
287  PyDict_SetItemString(gate, "type", type );
288  PyDict_SetItemString(gate, "target_qbit", target_qbit );
289  PyDict_SetItemString(gate, "control_qbit", control_qbit );
290 
291  Py_XDECREF(type);
292  Py_XDECREF(target_qbit);
293  Py_XDECREF(control_qbit);
294 
295  }
296  else if (operation->get_type() == U3_OPERATION) {
297 
298  // get U3 parameters
299  U3* u3_operation = static_cast<U3*>(operation);
300  double* parameters = (double*)malloc(3*sizeof(double));
301  u3_operation->get_optimized_parameters(parameters);
302 
303 
304  // create gate parameters
305  PyObject* type = Py_BuildValue("s", "U3" );
306  PyObject* target_qbit = Py_BuildValue("i", operation->get_target_qbit() );
307  PyObject* Theta = Py_BuildValue("f", parameters[0] );
308  PyObject* Phi = Py_BuildValue("f", parameters[1] );
309  PyObject* Lambda = Py_BuildValue("f", parameters[2] );
310 
311 
312  PyDict_SetItemString(gate, "type", type );
313  PyDict_SetItemString(gate, "target_qbit", target_qbit );
314  PyDict_SetItemString(gate, "Theta", Theta );
315  PyDict_SetItemString(gate, "Phi", Phi );
316  PyDict_SetItemString(gate, "Lambda", Lambda );
317 
318  Py_XDECREF(type);
319  Py_XDECREF(target_qbit);
320  Py_XDECREF(Theta);
321  Py_XDECREF(Phi);
322  Py_XDECREF(Lambda);
323 
324 
325  free( parameters);
326  }
327  else {
328 
329  }
330 
331  return gate;
332 
333 }

655 static PyObject *
656 qgd_N_Qubit_Decomposition_Wrapper_Reorder_Qubits(qgd_N_Qubit_Decomposition_Wrapper *self, PyObject *args ) {
657 
658  // initiate variables for input arguments
659  PyObject* qbit_list;
660 
661  // parsing input arguments
662  if (!PyArg_ParseTuple(args, "|O", &qbit_list )) return Py_BuildValue("i", -1);
663 
664  bool is_tuple = PyTuple_Check(qbit_list);
665  bool is_list = PyList_Check(qbit_list);
666 
667  // Check whether input is dictionary
668  if (!is_list && !is_tuple) {
669  printf("Input must be tuple or list!\n");
670  return Py_BuildValue("i", -1);
671  }
672 
673  // get the number of qbubits
674  Py_ssize_t element_num;
675 
676  if (is_tuple) {
677  element_num = PyTuple_GET_SIZE(qbit_list);
678  }
679  else {
680  element_num = PyList_GET_SIZE(qbit_list);
681  }
682 
683 
684  // create C++ variant of the tuple/list
685  std::vector<int> qbit_list_C( (int) element_num);
686  for ( Py_ssize_t idx=0; idx<element_num; idx++ ) {
687  if (is_tuple) {
688  qbit_list_C[(int) idx] = (int) PyLong_AsLong( PyTuple_GetItem(qbit_list, idx ) );
689  }
690  else {
691  qbit_list_C[(int) idx] = (int) PyLong_AsLong( PyList_GetItem(qbit_list, idx ) );
692  }
693 
694  }
695 
696 
697  // reorder the qubits in the decomposition class
698  self->decomp->reorder_qubits( qbit_list_C );
699 
700 
701 
702  return Py_BuildValue("i", 0);
