140 std::vector<Matrix> Gates_block::get_matrices( const double* parameters ) {

144 
145  for(std::vector<Gate*>::iterator it = gates.begin(); it != gates.end(); ++it) {
146 
147  Gate* operation = *it;
148  Matrix operation_mtx;
149 
150  if (operation->get_type() == CNOT_OPERATION) {
151  CNOT* cnot_operation = static_cast<CNOT*>(operation);
152  operation_mtx = cnot_operation->get_matrix();
153 
154  }
155  else if (operation->get_type() == CZ_OPERATION) {
156  CZ* cz_operation = static_cast<CZ*>(operation);
157  operation_mtx = cz_operation->get_matrix();
158  }
159  else if (operation->get_type() == CH_OPERATION) {
160  CH* ch_operation = static_cast<CH*>(operation);
161  operation_mtx = ch_operation->get_matrix();
162  }
163  else if (operation->get_type() == U3_OPERATION) {
164  U3* u3_operation = static_cast<U3*>(operation);
165 
166  if (u3_operation->get_parameter_num() == 1 ) {
167  operation_mtx = u3_operation->get_matrix( parameters );
168  parameters = parameters + 1;
169  }
170  else if (u3_operation->get_parameter_num() == 2 ) {
171  #ifdef DEBUG
172  if (isnan(parameters[0]) || isnan(parameters[1]) ) {
173  std::cout << "Gates_block::get_matrices: parameters contains NaN." << std::endl;
174  }
175 #endif
176  operation_mtx = u3_operation->get_matrix( parameters );
177  parameters = parameters + 2;
178  }
179  else if (u3_operation->get_parameter_num() == 3 ) {
180  operation_mtx = u3_operation->get_matrix( parameters );
181  parameters = parameters + 3;
182  }
183  else {
184  printf("The U3 operation has wrong number of parameters");
185  throw "The U3 operation has wrong number of parameters";
186  }
187 
188  }
189  else if (operation->get_type() == GENERAL_OPERATION) {
190  operation_mtx = operation->get_matrix();

192

195  std::cout << "Gates_block::get_matrices: operation_mtx contains NaN." << std::endl;

491 void Gates_block::list_gates( const double* parameters, int start_index ) {
492 
493  printf( "\nThe gates in the list of gates:\n" );
494 
495  int gate_idx = start_index;
496  int parameter_idx = parameter_num;
497 
498  for(int op_idx = gates.size()-1; op_idx>=0; op_idx--) {
499 
500  Gate* gate = gates[op_idx];
501 
502  if (gate->get_type() == CNOT_OPERATION) {
503  CNOT* cnot_gate = static_cast<CNOT*>(gate);
504 
505  printf( "%dth gate: CNOT with control qubit: %d and target qubit: %d\n", gate_idx, cnot_gate->get_control_qbit(), cnot_gate->get_target_qbit() );
506  gate_idx = gate_idx + 1;
507  }
508  else if (gate->get_type() == CZ_OPERATION) {
509  CZ* cz_gate = static_cast<CZ*>(gate);
510 
511  printf( "%dth gate: CZ with control qubit: %d and target qubit: %d\n", gate_idx, cz_gate->get_control_qbit(), cz_gate->get_target_qbit() );
512  gate_idx = gate_idx + 1;
513  }
514  else if (gate->get_type() == CH_OPERATION) {
515  CH* ch_gate = static_cast<CH*>(gate);
516 
517  printf( "%dth gate: CH with control qubit: %d and target qubit: %d\n", gate_idx, ch_gate->get_control_qbit(), ch_gate->get_target_qbit() );
518  gate_idx = gate_idx + 1;
519  }
520  else if (gate->get_type() == U3_OPERATION) {
521 
522  // definig the U3 parameters
523  double vartheta;
524  double varphi;
525  double varlambda;
526 
527  // get the inverse parameters of the U3 rotation
528 
529  U3* u3_gate = static_cast<U3*>(gate);
530 
531  if ((u3_gate->get_parameter_num() == 1) && u3_gate->is_theta_parameter()) {
532  vartheta = std::fmod( parameters[parameter_idx-1], 4*M_PI);
533  varphi = 0;
534  varlambda =0;
535  parameter_idx = parameter_idx - 1;
536 
537  }
538  else if ((u3_gate->get_parameter_num() == 1) && u3_gate->is_phi_parameter()) {
539  vartheta = 0;
540  varphi = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
541  varlambda =0;
542  parameter_idx = parameter_idx - 1;
543  }
544  else if ((u3_gate->get_parameter_num() == 1) && u3_gate->is_lambda_parameter()) {
545  vartheta = 0;
546  varphi = 0;
547  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
548  parameter_idx = parameter_idx - 1;
549  }
550  else if ((u3_gate->get_parameter_num() == 2) && u3_gate->is_theta_parameter() && u3_gate->is_phi_parameter() ) {
551  vartheta = std::fmod( parameters[ parameter_idx-2 ], 4*M_PI);
552  varphi = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
553  varlambda = 0;
554  parameter_idx = parameter_idx - 2;
555  }
556  else if ((u3_gate->get_parameter_num() == 2) && u3_gate->is_theta_parameter() && u3_gate->is_lambda_parameter() ) {
557  vartheta = std::fmod( parameters[ parameter_idx-2 ], 4*M_PI);
558  varphi = 0;
559  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
560  parameter_idx = parameter_idx - 2;
561  }
562  else if ((u3_gate->get_parameter_num() == 2) && u3_gate->is_phi_parameter() && u3_gate->is_lambda_parameter() ) {
563  vartheta = 0;
564  varphi = std::fmod( parameters[ parameter_idx-2], 2*M_PI);
565  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
566  parameter_idx = parameter_idx - 2;
567  }
568  else if ((u3_gate->get_parameter_num() == 3)) {
569  vartheta = std::fmod( parameters[ parameter_idx-3 ], 4*M_PI);
570  varphi = std::fmod( parameters[ parameter_idx-2 ], 2*M_PI);
571  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
572  parameter_idx = parameter_idx - 3;
573  }
574 
575 // message = message + "U3 on target qubit %d with parameters theta = %f, phi = %f and lambda = %f";
576  printf("%dth gate: U3 on target qubit: %d and with parameters theta = %f, phi = %f and lambda = %f\n", gate_idx, u3_gate->get_target_qbit(), vartheta, varphi, varlambda );
577  gate_idx = gate_idx + 1;
578 
579  }
580  else if (gate->get_type() == BLOCK_OPERATION) {
581  Gates_block* block_gate = static_cast<Gates_block*>(gate);
582  const double* parameters_layer = parameters + parameter_idx - gate->get_parameter_num();
583  block_gate->list_gates( parameters_layer, gate_idx );
584  parameter_idx = parameter_idx - block_gate->get_parameter_num();
585  gate_idx = gate_idx + block_gate->get_gate_num();
586  }
587 
588  }
