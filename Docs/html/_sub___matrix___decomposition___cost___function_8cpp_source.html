182 void functor_submtx_cost_fnc::operator()( int product_idx ) const {
183 
184  // number of elements in the matrix of submatrix products
185  int element_num = submatrix_size*submatrix_size;
186 
187  // preallocate memeory for submatrix product
188  QGD_Complex16* submatrix_prod = (QGD_Complex16*)qgd_calloc(element_num,sizeof(QGD_Complex16), 64);
189 
190  // number of submatrices
191  int submatrices_num_row = 2;
192 
193  // select the given submatrices used to calculate the partial cost_function
194  int jdx = product_idx % submatrices_num_row;
195  int idx = (int) ( product_idx - jdx )/submatrices_num_row;
196 
197  // calculate the submatrix product
198  zgemm3m_wrapper_adj( submatrices[idx], submatrices[jdx], submatrix_prod, submatrix_size);
199 
200 
201  // subtract the corner element from the diagonal
202  QGD_Complex16 corner_element = submatrix_prod[0];
203  for ( int row_idx=0; row_idx<submatrix_size; row_idx++) {
204  int element_idx = row_idx*submatrix_size+row_idx;
205  submatrix_prod[element_idx].real = submatrix_prod[element_idx].real - corner_element.real;
206  submatrix_prod[element_idx].imag = submatrix_prod[element_idx].imag - corner_element.imag;
207  }
208 
209  // Calculate the |x|^2 value of the elements of the submatrixproducts
210  for ( int idx=0; idx<element_num; idx++ ) {
211  submatrix_prod[idx].real = submatrix_prod[idx].real*submatrix_prod[idx].real + submatrix_prod[idx].imag*submatrix_prod[idx].imag;
212  // for performance reason we leave the imaginary part intact (we dont neet it anymore)
213  //submatrix_prods[idx].imag = 0;
214  }
215 
216 
217 
218  // summing up elements and calculate the final cost function
219  double cost_function_partial = 0;
220  for ( int row_idx=0; row_idx<submatrix_size; row_idx++ ) {
221 
222  // calculate the sum for each row
223  for (int col_idx=0; col_idx<submatrix_size; col_idx++) {
224  int element_idx = row_idx*submatrix_size + col_idx;
225  cost_function_partial = cost_function_partial + submatrix_prod[element_idx].real;
226  }
227  }
228 
229 
230  // checking NaN
231  if (std::isnan(cost_function_partial)) {
232  printf("cost function NaN on thread %d: exiting\n", product_idx);
233  exit(-1);
234  }
235 
236 
237  // store the calculated value for the given submatrix product
238  prod_cost_functions[product_idx] = cost_function_partial;
239 
240 
241 
242  // release submatrix product
243  if (submatrix_prod != NULL ) {
244  qgd_free( submatrix_prod );
245  submatrix_prod = NULL;
246  }
