161 void functor_submtx_cost_fnc::operator()( int product_idx ) const {
162 
163 
164 
165  // number of submatrices
166  size_t submatrices_num_row = 2;
167 
168  // select the given submatrices used to calculate the partial cost_function
169  size_t jdx = product_idx % submatrices_num_row;
170  size_t idx = (size_t) ( product_idx - jdx )/submatrices_num_row;
171 
172  // calculate the submatrix product
173  Matrix tmp = (*submatrices)[jdx];
174  tmp.transpose();
175  tmp.conjugate();
176  Matrix submatrix_prod = dot( (*submatrices)[idx], tmp);
177 
178 #ifdef DEBUG
179  if (submatrix_prod.isnan()) {
180  std::cout << "functor_submtx_cost_fnc::operator: Submatrix product contains NaN. Exiting" << std::endl;
181  }
182 #endif
183 
184 
185 
186  // number of elements in the matrix of submatrix products
187  size_t submatrix_size = submatrix_prod.rows;
188  size_t element_num = submatrix_size*submatrix_size;
189 
190  // subtract the corner element from the diagonal
191  QGD_Complex16 corner_element = submatrix_prod[0];
192  tbb::parallel_for( tbb::blocked_range<size_t>(0, submatrix_size, 1), [&](tbb::blocked_range<size_t> r){
193  for ( size_t row_idx=r.begin(); row_idx != r.end(); row_idx++) {
194  size_t element_idx = row_idx*submatrix_size+row_idx;
195  submatrix_prod[element_idx].real = submatrix_prod[element_idx].real - corner_element.real;
196  submatrix_prod[element_idx].imag = submatrix_prod[element_idx].imag - corner_element.imag;
197  }
198 
199  });
200 
201  // Calculate the |x|^2 value of the elements of the submatrixproducts and add to the partial cost function
202  tbb::parallel_for( tbb::blocked_range<size_t>(0, element_num, 1), [&](tbb::blocked_range<size_t> r){
203  for (size_t idx = r.begin(); idx != r.end(); idx ++) {
204 
205  // store the calculated value for the given submatrix product element
206  double &prod_cost_function_priv = prod_cost_functions->local();
207  prod_cost_function_priv = prod_cost_function_priv + submatrix_prod[idx].real*submatrix_prod[idx].real + submatrix_prod[idx].imag*submatrix_prod[idx].imag;
208  }
209  });
210 
211  // checking NaN
212  if (std::isnan(prod_cost_functions->local())) {
213  std::cout << "cost function NaN on cost function product "<< product_idx << std::endl;
214  }
215 
216 
217 }
