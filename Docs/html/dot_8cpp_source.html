45 dot( Matrix &A, Matrix &B ) {
46 
47 #if BLAS==0 // undefined BLAS
48  int NumThreads = omp_get_max_threads();
49  omp_set_num_threads(1);
50 #elif BLAS==1 // MKL
51  int NumThreads = mkl_get_max_threads();
52  MKL_Set_Num_Threads(1);
53 #elif BLAS==2 //OpenBLAS
54  int NumThreads = openblas_get_num_threads();
55  openblas_set_num_threads(1);
56 #endif
57 
58 
59  // check the Matrix shapes in DEBUG mode
60  assert( check_matrices( A, B ) );
61 
62 
63 #if BLAS==1 // MKL does not support option CblasConjNoTrans so the conjugation of the matrices are done in prior.
64  if ( B.is_conjugated() && !B.is_transposed() ) {
65  Matrix tmp = Matrix( B.rows, B.cols );
66  vzConj( B.cols*B.rows, B.get_data(), tmp.get_data() );
67  B = tmp;
68  }
69 
70  if ( A.is_conjugated() && !A.is_transposed() ) {
71  Matrix tmp = Matrix( A.rows, A.cols );
72  vzConj( A.cols*A.rows, A.get_data(), tmp.get_data() );
73  A = tmp;
74  }
75 #endif
76 
77 
78  // Preparing the output Matrix
79  Matrix C;
80  if ( A.is_transposed() ){
81  if ( B.is_transposed() ) {
82  C = Matrix(A.cols, B.rows);
83  }
84  else {
85  C = Matrix(A.cols, B.cols);
86  }
87  }
88  else {
89  if ( B.is_transposed() ) {
90  C = Matrix(A.rows, B.rows);
91  }
92  else {
93  C = Matrix(A.rows, B.cols);
94  }
95  }
96 
97  // Calculating the matrix product
98  if ( A.rows <= SERIAL_CUTOFF && B.cols <= SERIAL_CUTOFF ) {
99  // creating the serial task object
100  zgemm3m_Task_serial calc_task = zgemm3m_Task_serial( A, B, C );
101  calc_task.zgemm3m_chunk();
102  }
103  else {
104  // creating the task object
105  zgemm3m_Task& calc_task = *new(tbb::task::allocate_root()) zgemm3m_Task( A, B, C );
106 
107  // starting parallel calculations
108  tbb::task::spawn_root_and_wait(calc_task);
109  }
110 
111 #if BLAS==0 // undefined BLAS
112  omp_set_num_threads(NumThreads);
113 #elif BLAS==1 //MKL
114  MKL_Set_Num_Threads(NumThreads);
115 #elif BLAS==2 //OpenBLAS
116  openblas_set_num_threads(NumThreads);
117 #endif
118 
119  return C;

133 check_matrices( Matrix &A, Matrix &B ) {
134 
135 
136  if (!A.is_transposed() & !B.is_transposed()) {
137  if ( A.cols != B.rows ) {
138  std::cout << "pic::dot:: Cols of matrix A does not match rows of matrix B!" << std::endl;
139  return false;
140  }
141  }
142  else if ( A.is_transposed() & !B.is_transposed() ) {
143  if ( A.rows != B.rows ) {
144  std::cout << "pic::dot:: Cols of matrix A.transpose does not match rows of matrix B!" << std::endl;
145  return false;
146  }
147  }
148  else if ( A.is_transposed() & B.is_transposed() ) {
149  if ( A.rows != B.cols ) {
150  std::cout << "pic::dot:: Cols of matrix A.transpose does not match rows of matrix B.transpose!" << std::endl;
151  return false;
152  }
153  }
154  else if ( !A.is_transposed() & B.is_transposed() ) {
155  if ( A.cols != B.cols ) {
156  std::cout << "pic::dot:: Cols of matrix A does not match rows of matrix B.transpose!" << std::endl;
157  return false;
158  }
159  }
160 
161 
162  // check the pointer of the matrices
163  if ( A.get_data() == NULL ) {
164  std::cout << "pic::dot:: No preallocated data in matrix A!" << std::endl;
165  return false;
166  }
167  if ( B.get_data() == NULL ) {
168  std::cout << "pic::dot:: No preallocated data in matrix B!" << std::endl;
169  return false;
170  }
171 
172  return true;
