211 
212  // get the most unitary submatrix
213  // get the number of 2qubit submatrices
214  int submatrices_num_row = 2;
215 
216  // get the size of the submatrix
217  int submatrix_size = int(matrix_size/2);
218 
219  // fill up the submatrices and select the most unitary submatrix
220 
221  Matrix most_unitary_submatrix_mtx = Matrix(submatrix_size, submatrix_size );
222  QGD_Complex16* most_unitary_submatrix = most_unitary_submatrix_mtx.get_data();
223  double unitary_error_min = 1e8;
224 
225  for (int idx=0; idx<submatrices_num_row; idx++) { // in range(0,submatrices_num_row):
226  for (int jdx=0; jdx<submatrices_num_row; jdx++) { // in range(0,submatrices_num_row):
227 
228  Matrix submatrix_mtx = Matrix(submatrix_size, submatrix_size);
229  QGD_Complex16* submatrix = submatrix_mtx.get_data();
230 
231  for ( int row_idx=0; row_idx<submatrix_size; row_idx++ ) {
232  int matrix_offset = idx*(matrix_size*submatrix_size) + jdx*(submatrix_size) + row_idx*matrix_size;
233  int submatrix_offset = row_idx*submatrix_size;
234  memcpy(submatrix+submatrix_offset, subdecomposed_matrix+matrix_offset, submatrix_size*sizeof(QGD_Complex16));
235  }
236 
237  // calculate the product of submatrix*submatrix'
238  Matrix submatrix_mtx_adj = submatrix_mtx;
239  submatrix_mtx_adj.transpose();
240  submatrix_mtx_adj.conjugate();
241  Matrix submatrix_prod = dot( submatrix_mtx, submatrix_mtx_adj);
242 
243  // subtract corner element
244  QGD_Complex16 corner_element = submatrix_prod[0];
245  for (int row_idx=0; row_idx<submatrix_size; row_idx++) {
246  submatrix_prod[row_idx*submatrix_size+row_idx].real = submatrix_prod[row_idx*submatrix_size+row_idx].real - corner_element.real;
247  submatrix_prod[row_idx*submatrix_size+row_idx].imag = submatrix_prod[row_idx*submatrix_size+row_idx].imag - corner_element.imag;
248  }
249 
250  double unitary_error = cblas_dznrm2( submatrix_size*submatrix_size, submatrix_prod.get_data(), 1 );
251 
252  if (unitary_error < unitary_error_min) {
253  unitary_error_min = unitary_error;
254  memcpy(most_unitary_submatrix, submatrix, submatrix_size*submatrix_size*sizeof(QGD_Complex16));
255  }
256 
257  }
258  }
259 
260

262 
263  // create class tp decompose submatrices
264  N_Qubit_Decomposition<subdecomp_class>* cdecomposition = new N_Qubit_Decomposition<subdecomp_class>(most_unitary_submatrix_mtx, qbit_num-1, optimize_layer_num, initial_guess);

275

569 int simplify_layer( Operation_block* layer, double* parameters, unsigned int parameter_num_block, std::map<int,int> max_layer_num_loc, Operation_block* &simplified_layer, double* &simplified_parameters, unsigned int &simplified_parameter_num) {
570 
571  if (verbose) {
572  printf("Try to simplify sub-structure \n");
573  }
574 
575  // get the target bit
576  int target_qbit = -1;
577  int control_qbit = -1;
578 
579  std::vector<Operation*> layer_operations = layer->get_operations();
580  for (std::vector<Operation*>::iterator it = layer_operations.begin(); it!=layer_operations.end(); it++) {
581  Operation* op = *it;
582  if (op->get_type() == CNOT_OPERATION) {
583  target_qbit = op->get_target_qbit();
584  control_qbit = op->get_control_qbit();
585  break;
586  }
587  }
588 
589  // if there are no target or control qubits, return the initial values
590  if ( (target_qbit == -1) || ( control_qbit == -1 ) ) {
591  return 1;
592  }
593 
594  // get the matrix of the two qubit space
595 
596  // reorder the control and target qubits to the end of the list
597  std::vector<int> qbits_reordered;
598  for (int qbit_idx=qbit_num-1; qbit_idx>-1; qbit_idx-- ) { // in range(self.qbit_num-1,-1,-1):
599  if ( (qbit_idx != target_qbit) && (qbit_idx != control_qbit) ) {
600  qbits_reordered.push_back(qbit_idx);
601  }
602  }
603 
604  qbits_reordered.push_back(control_qbit);
605  qbits_reordered.push_back(target_qbit);
606 
607  // construct abstarct operations correspond to the reordeerd qubits
608  Operation_block* reordered_layer = layer->clone();
609  reordered_layer->reorder_qubits( qbits_reordered );
610 
611  // get the reordered N-qubit matrix of the layer
612  Matrix full_matrix_reordered = reordered_layer->get_matrix( parameters );
613  delete reordered_layer;
614 
615 
616 
617 
618 
619 
620  // construct the Two-qubit submatrix from the reordered matrix
621  Matrix submatrix = Matrix(4,4);
622  for ( int element_idx=0; element_idx<16; element_idx++) {
623  int col_idx = element_idx % 4;
624  int row_idx = int((element_idx-col_idx)/4);
625  submatrix[element_idx].real = full_matrix_reordered[col_idx*matrix_size+row_idx].real;
626  submatrix[element_idx].imag = -full_matrix_reordered[col_idx*matrix_size+row_idx].imag;
627  }
628 
629  // decompose the chosen 2-qubit unitary
630  N_Qubit_Decomposition<subdecomp_class>* cdecomposition = new N_Qubit_Decomposition<subdecomp_class>(submatrix, 2, true, initial_guess);
631 
632  // set the maximal number of layers
633  cdecomposition->set_max_layer_num( max_layer_num_loc );
634 
635  // suppress output messages
636  cdecomposition->set_verbose( false );
637 
638  // starting the decomposition
639  cdecomposition->start_decomposition(true, false);
640 
641 
642 
643  // check whether simplification was succesfull
644  if (!cdecomposition->check_optimization_solution()) {
645  // return with the original layer, if the simplification wa snot successfull
646  if (verbose) {
647  printf("The simplification of the sub-structure was not possible\n");
648  }
649  delete cdecomposition;
650  return 1;
651  }

654 
655  // contruct the layer containing the simplified operations in the N-qubit space
656  // but first get the inverse reordered qubit list
657  std::vector<int> qbits_inverse_reordered;
658  for (int idx=0; idx<qbit_num; idx++) {
659  qbits_inverse_reordered.push_back(-1);
660  }
661 
662  for (int qbit_idx=qbit_num-1; qbit_idx>-1; qbit_idx-- ) { // in range(self.qbit_num-1,-1,-1):
663  qbits_inverse_reordered[qbit_num-1-qbits_reordered[qbit_num-1-qbit_idx]] = qbit_idx;
664  }

666  simplified_layer = new Operation_block(qbit_num);

669  for ( std::vector<Operation*>::iterator it=simplifying_operations.begin(); it!=simplifying_operations.end(); it++) { //int operation_block_idx=0 in range(0,len(cdecomposition.operations)):
670  Operation* op = *it;
671 
672  Operation_block* block_op = static_cast<Operation_block*>( op );
673  block_op->set_qbit_num( qbit_num );
674  block_op->reorder_qubits( qbits_inverse_reordered );
675  simplified_layer->combine( block_op );
676  }

678
