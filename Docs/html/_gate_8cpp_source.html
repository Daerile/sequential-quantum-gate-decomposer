173  int control_qbit_new = control_qbit;

176  // setting the new value for the target qubit
177  for (int idx=0; idx<qbit_num; idx++) {
178  if (target_qbit == qbit_list[idx]) {

180  }

182  control_qbit_new = qbit_num-1-idx;
183  }
184  }
