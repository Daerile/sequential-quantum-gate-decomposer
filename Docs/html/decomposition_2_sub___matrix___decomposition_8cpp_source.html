119 void Sub_Matrix_Decomposition::disentangle_submatrices() {
120 
121  if (subdisentaglement_done) {
122  if (verbose) {
123  printf("Sub-disentaglement already done.\n");
124  }
125  return;
126  }
127 
128  if (verbose) {
129  printf("\nDisentagling submatrices.\n");
130  }
131 
132  // setting the global target minimum
133  global_target_minimum = 0;
134  current_minimum = optimization_problem(NULL);
135 
136  // check if it needed to do the subunitarization
137  if (check_optimization_solution()) {
138  if (verbose) {
139  printf("Disentanglig not needed\n");
140  }
141  subdecomposed_mtx = Umtx;
142  subdisentaglement_done = true;
143  return;
144  }
145 
146 
147 
148  if ( !check_optimization_solution() ) {
149  // Adding the gates of the successive layers
150 
151  //measure the time for the decompositin
152  clock_t start_time = time(NULL);
153 
154  // the maximal number of layers in the subdeconposition
155  int max_layer_num_loc;
156  try {
157  max_layer_num_loc = max_layer_num[qbit_num];
158  }
159  catch (...) {
160  throw "Layer number not given";
161  }
162 
163  while ( layer_num < max_layer_num_loc ) {
164 
165  // add another gate layers to the gate structure used in the decomposition
166  add_gate_layers();
167 
168  // get the number of blocks
169  layer_num = gates.size();
170 
171  // Do the optimization
172  if (optimize_layer_num || layer_num >= max_layer_num_loc ) {
173 
174  // release optimized parameters if necessary
175  if (optimized_parameters != NULL) {
176  qgd_free(optimized_parameters);
177  optimized_parameters = NULL;
178  }
179 
180  // solve the optimization problem to find the correct minimum
181  solve_optimization_problem( NULL, 0);
182 
183 
184  if (check_optimization_solution()) {
185  break;
186  }
187  }
188 
189  }
190 
191  if (verbose) {
192  printf("--- %f seconds elapsed during the decomposition ---\n\n", float(time(NULL) - start_time));
193  }
194 
195 
196  }
197 
198 
199 
200  if (check_optimization_solution()) {
201  if (verbose) {
202  printf("Sub-disentaglement was succesfull.\n\n");
203  }
204  }
205  else {
206  if (verbose) {
207  printf("Sub-disentaglement did not reach the tolerance limit.\n\n");
208  }
209  }
210 
211 
212  // indicate that the unitarization of the sumbatrices was done
213  subdisentaglement_done = true;
214 
215  // The subunitarized matrix
216  subdecomposed_mtx = get_transformed_matrix( optimized_parameters, gates.begin(), gates.size(), Umtx );

222 void Sub_Matrix_Decomposition::add_gate_layers() {
223 
224  if ( unit_gate_structure == NULL ) {
225  // add default gate structure if custom is not given
226  add_default_gate_layers();
227  return;
228  }
229 
231  // add custom gate structure
232 
233  // the number of succeeding identical layers in the subdecomposition
234  int identical_blocks_loc;
235  try {
236  identical_blocks_loc = identical_blocks[qbit_num];
237  if (identical_blocks_loc==0) {
238  identical_blocks_loc = 1;
239  }
240  }
241  catch (...) {
242  identical_blocks_loc=1;
243  }
244 
245  // get the list of sub-blocks in the custom gate structure
246  std::vector<Gate*> gates = unit_gate_structure->get_gates();
247 
248  for (std::vector<Gate*>::iterator gates_it = gates.begin(); gates_it != gates.end(); gates_it++ ) {
249 
250  // The current gate
251  Gate* gate = *gates_it;
252 
253  for (int idx=0; idx<identical_blocks_loc; idx++) {
254  if (gate->get_type() == CNOT_OPERATION ) {
255  CNOT* cnot_gate = static_cast<CNOT*>( gate );
256  add_gate_to_end( (Gate*)cnot_gate->clone() );
257  }
258  else if (gate->get_type() == CZ_OPERATION ) {
259  CZ* cz_gate = static_cast<CZ*>( gate );
260  add_gate_to_end( (Gate*)cz_gate->clone() );
261  }
262  else if (gate->get_type() == GENERAL_OPERATION ) {
263  add_gate_to_end( gate->clone() );
264  }
265  else if (gate->get_type() == U3_OPERATION ) {
266  U3* u3_gate = static_cast<U3*>( gate );
267  add_gate_to_end( (Gate*)u3_gate->clone() );
268  }
269  else if (gate->get_type() == BLOCK_OPERATION ) {
270  Gates_block* block_gate = static_cast<Gates_block*>( gate );
271  add_gate_to_end( (Gate*)block_gate->clone() );
272  }
273 
274  }
275  }

350 void Sub_Matrix_Decomposition::solve_layer_optimization_problem( int num_of_parameters, gsl_vector *solution_guess_gsl) {
351 
352 
353  if (gates.size() == 0 ) {
354  return;
355  }
356 
357 
358  if (solution_guess_gsl == NULL) {
359  solution_guess_gsl = gsl_vector_alloc(num_of_parameters);
360  }
361 
362  if (optimized_parameters == NULL) {
363  optimized_parameters = (double*)qgd_calloc(num_of_parameters,sizeof(double), 64);
364  memcpy(optimized_parameters, solution_guess_gsl->data, num_of_parameters*sizeof(double) );
365  }
366 
367  // maximal number of iteration loops
368  int iteration_loops_max;
369  try {
370  iteration_loops_max = std::max(iteration_loops[qbit_num], 1);
371  }
372  catch (...) {
373  iteration_loops_max = 1;
374  }
375 
376 
377  // do the optimization loops
378  for (int idx=0; idx<iteration_loops_max; idx++) {
379 
380  size_t iter = 0;
381  int status;
382 
383  const gsl_multimin_fdfminimizer_type *T;
384  gsl_multimin_fdfminimizer *s;
385 
386  Sub_Matrix_Decomposition* par = this;
387 
388 
389  gsl_multimin_function_fdf my_func;
390 
391 
392  my_func.n = num_of_parameters;
393  my_func.f = optimization_problem;
394  my_func.df = optimization_problem_grad;
395  my_func.fdf = optimization_problem_combined;
396  my_func.params = par;
397 
398 
399  T = gsl_multimin_fdfminimizer_vector_bfgs2;
400  s = gsl_multimin_fdfminimizer_alloc (T, num_of_parameters);
401 
402 
403  gsl_multimin_fdfminimizer_set (s, &my_func, solution_guess_gsl, 0.1, 0.1);
404 
405  do {
406  iter++;
407 
408  status = gsl_multimin_fdfminimizer_iterate (s);
409 
410  if (status) {
411  break;
412  }
413 
414  status = gsl_multimin_test_gradient (s->gradient, 1e-1);
415  /*if (status == GSL_SUCCESS) {
416  printf ("Minimum found\n");
417  }*/
418 
419  } while (status == GSL_CONTINUE && iter < 100);
420 
421  if (current_minimum > s->f) {
422  current_minimum = s->f;
423  memcpy( optimized_parameters, s->x->data, num_of_parameters*sizeof(double) );
424  gsl_multimin_fdfminimizer_free (s);
425 
426  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
427  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
428  }
429  }
430  else {
431  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
432  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
433  }
434 
435  gsl_multimin_fdfminimizer_free (s);
436  }
437 
438 
439 
440  }
