35 // @brief Call to create a random unitary containing a given number of CNOT gates between randomly chosen qubits
36 // @param qbit_num The number of qubits spanning the unitary
37 // @param cnot_num The number of CNOT gates in the unitary
38 void few_CNOT_unitary( int qbit_num, int cnot_num, QGD_Complex16* mtx) {
39 
40  // the current number of CNOT gates
41  int cnot_num_curr = 0;
42 
43  // the size of the matrix
44  int matrix_size = Power_of_2(qbit_num);
45 
46  // The unitary discribing each qubits in their initial state
47  memset( mtx, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );
48  create_identity( mtx, matrix_size );
49 
50  QGD_Complex16* mtx_tmp = (QGD_Complex16*)qgd_calloc( matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
51 
52  QGD_Complex16* gate_matrix = (QGD_Complex16*)qgd_calloc( matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
53  memset( gate_matrix, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );
54 
55  // constructing the unitary
56  while (true) {
57  int cnot_or_u3 = rand() % 5 + 1;
58 
59  CNOT* cnot_op = NULL;
60  U3* u3_op = NULL;
61 
62  if (cnot_or_u3 <= 4) {
63  // creating random parameters for the U3 operation
64  double parameters[3];
65 
66  parameters[0] = double(rand())/RAND_MAX*4*M_PI;
67  parameters[1] = double(rand())/RAND_MAX*2*M_PI;
68  parameters[2] = double(rand())/RAND_MAX*2*M_PI;
69 
70 
71  // randomly choose the target qbit
72  int target_qbit = rand() % qbit_num;
73 
74  // creating the U3 gate
75  u3_op = new U3(qbit_num, target_qbit, true, true, true);
76 
77  // get the matrix of the operation
78  u3_op->matrix(parameters, gate_matrix);
79  }
80  else if ( cnot_or_u3 == 5 ) {
81  // randomly choose the target qbit
82  int target_qbit = rand() % qbit_num;
83 
84  // randomly choose the control qbit
85  int control_qbit = rand() % qbit_num;
86 
87  if (target_qbit == control_qbit) {
88  create_identity( gate_matrix, matrix_size );
89  }
90  else {
91 
92  // creating the CNOT gate
93  cnot_op = new CNOT(qbit_num, control_qbit, target_qbit);
94 
95  // get the matrix of the operation
96  cnot_op->matrix(gate_matrix);
97 
98  cnot_num_curr = cnot_num_curr + 1;
99  }
100  }
101  else {
102  create_identity( gate_matrix, matrix_size );
103  }
104 
105 
106  // get the current unitary
107  zgemm3m_wrapper(gate_matrix, mtx, mtx_tmp, matrix_size);
108  memset( gate_matrix, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );
109  memcpy( mtx, mtx_tmp, matrix_size*matrix_size*sizeof(QGD_Complex16) );
110 
111 
112  delete u3_op;
113  u3_op = NULL;
114 
115  delete cnot_op;
116  cnot_op = NULL;
117 
118 
119 
120 
121  // exit the loop if the maximal number of CNOT gates reched
122  if (cnot_num_curr >= cnot_num) {
123 
124  if (mtx_tmp != NULL) {
125  qgd_free( mtx_tmp );
126  }
127 
128  if (gate_matrix != NULL) {
129  qgd_free( gate_matrix );
130  }
131 
132  return;
133  }

135  }
