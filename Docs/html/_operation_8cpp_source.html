138 void Operation::reorder_qubits( vector<int> qbit_list ) {
139 
140  // check the number of qubits
141  if ((int)qbit_list.size() != qbit_num ) {
142  printf("Wrong number of qubits\n");
143  exit(-1);
144  }
145 
146 
147  int control_qbit_new = control_qbit;
148  int target_qbit_new = target_qbit;
149 
150  // setting the new value for the target qubit
151  for (int idx=0; idx<qbit_num; idx++) {
152  if (target_qbit == qbit_list[idx]) {
153  target_qbit_new = qbit_num-1-idx;
154  }
155  if (control_qbit == qbit_list[idx]) {
156  control_qbit_new = qbit_num-1-idx;
157  }
158  }
159 
160  control_qbit = control_qbit_new;
161  target_qbit = target_qbit_new;
162 }
