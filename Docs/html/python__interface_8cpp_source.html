40 void* iface_new_N_Qubit_Decomposition( double* mtx_real, double* mtx_imag, int qbit_num, bool optimize_layer_num, int initial_guess_num ) {
41 
42  int matrix_size = Power_of_2(qbit_num);
43 
44  // combining real and imaginary parts of the matrix inti MKL complex matrix
45  int element_num = matrix_size*matrix_size;
46  QGD_Complex16* Umtx = (QGD_Complex16*)qgd_calloc(element_num,sizeof(QGD_Complex16), 64);
47 
48  #pragma omp parallel for
49  for(int idx = 0; idx < element_num; idx++) {
50  Umtx[idx].real = mtx_real[idx];
51  Umtx[idx].imag = mtx_imag[idx];
52  }
53 
54  // setting the initial guess type
55  guess_type initial_guess;
56  if ( initial_guess_num==0 ) {
57  initial_guess = ZEROS;
58  }
59  else if ( initial_guess_num==1 ) {
60  initial_guess = RANDOM;
61  }
62  else if ( initial_guess_num==2 ) {
63  initial_guess = CLOSE_TO_ZERO;
64  }
65  else {
66  printf("Wrong initial guess\n");
67  exit(-1);
68  }
69 
70 
71  // creating an instance of class N_Qubit_decomposition
72  N_Qubit_Decomposition* instance = new N_Qubit_Decomposition( Umtx, qbit_num, optimize_layer_num, initial_guess );
73 
74  return (void*)instance;
