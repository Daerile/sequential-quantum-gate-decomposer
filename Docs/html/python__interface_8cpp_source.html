37 void* iface_new_N_Qubit_Decomposition( double* mtx_real, double* mtx_imag, int qbit_num, bool optimize_layer_num, int initial_guess_num ) {
38 
39  int matrix_size = Power_of_2(qbit_num);
40 
41  // combining real and imaginary parts of the matrix inti MKL complex matrix
42  int element_num = matrix_size*matrix_size;
43  QGD_Complex16* Umtx = (QGD_Complex16*)qgd_calloc(element_num,sizeof(QGD_Complex16), 64);
44 
45  #pragma omp parallel for
46  for(int idx = 0; idx < element_num; idx++) {
47  Umtx[idx].real = mtx_real[idx];
48  Umtx[idx].imag = mtx_imag[idx];
49  }
50 
51 
52  // skeleton input variables to initialize the class
53  std::map<int,int> num_of_layers;
54  std::map<int,int> identical_blocks;
55 
56  // setting the initial guess type
57  guess_type initial_guess;
58  if ( initial_guess_num==0 ) {
59  initial_guess = ZEROS;
60  }
61  else if ( initial_guess_num==1 ) {
62  initial_guess = RANDOM;
63  }
64  else if ( initial_guess_num==2 ) {
65  initial_guess = CLOSE_TO_ZERO;
66  }
67 
68 
69  // creating an instance of class N_Qubit_decomposition
70  N_Qubit_Decomposition* instance = new N_Qubit_Decomposition( Umtx, qbit_num, num_of_layers, identical_blocks, optimize_layer_num, initial_guess );
