294 static int
295 qgd_N_Qubit_Decomposition_adaptive_Wrapper_init(qgd_N_Qubit_Decomposition_adaptive_Wrapper *self, PyObject *args, PyObject *kwds)
296 {
297  // The tuple of expected keywords
298  static char *kwlist[] = {(char*)"Umtx", (char*)"qbit_num", (char*)"level_limit", (char*)"level_limit_min", (char*)"method", (char*)"topology", NULL};
299 
300  // initiate variables for input arguments
301  PyObject *Umtx_arg = NULL;
302  int qbit_num = -1;
303  int level_limit = 0;
304  int level_limit_min = 0;
305  PyObject *method = NULL;
306  PyObject *topology = NULL;
307 
308  // parsing input arguments
309  if (!PyArg_ParseTupleAndKeywords(args, kwds, "|OiiiOO", kwlist,
310  &Umtx_arg, &qbit_num, &level_limit, &level_limit_min, &method, &topology))
311  return -1;
312 
313  // convert python object array to numpy C API array
314  if ( Umtx_arg == NULL ) return -1;
315  self->Umtx = PyArray_FROM_OTF(Umtx_arg, NPY_COMPLEX128, NPY_ARRAY_IN_ARRAY);
316 
317  // test C-style contiguous memory allocation of the array
318  if ( !PyArray_IS_C_CONTIGUOUS(self->Umtx) ) {
319  std::cout << "Umtx is not memory contiguous" << std::endl;
320  }
321 
322 
323  // create QGD version of the Umtx
324  Matrix Umtx_mtx = numpy2matrix(self->Umtx);
325 
326 
327 
328 
329  // determine the optimizaton method
330  PyObject* method_string = PyObject_Str(method);
331  PyObject* method_string_unicode = PyUnicode_AsEncodedString(method_string, "utf-8", "~E~");
332  const char* method_C = PyBytes_AS_STRING(method_string_unicode);
333 
334  // elaborate connectivity topology
335  bool is_None = topology == Py_None;
336  bool is_list = PyList_Check(topology);
337 
338  // Check whether input is a list
339  if (!is_list && !is_None) {
340  printf("Input topology must be a list!\n");
341  return -1;
342  }
343 
344  // create C++ variant of the list
345  std::vector<matrix_base<int>> topology_Cpp;
346 
347  if ( !is_None ) {
348 
349  // get the number of qbubits
350  Py_ssize_t element_num = PyList_GET_SIZE(topology);
351 
352  for ( Py_ssize_t idx=0; idx<element_num; idx++ ) {
353  PyObject *item = PyList_GetItem(topology, idx );
354 
355  // Check whether input is a list
356  if (!PyTuple_Check(item)) {
357  printf("Elements of topology must be a tuple!\n");
358  return -1;
359  }
360 
361  matrix_base<int> item_Cpp(1,2);
362  item_Cpp[0] = (int) PyLong_AsLong( PyTuple_GetItem(item, 0 ) );
363  item_Cpp[1] = (int) PyLong_AsLong( PyTuple_GetItem(item, 1 ) );
364 
365  topology_Cpp.push_back( item_Cpp );
366  }
367  }
368 
369 
370  // create an instance of the class N_Qubit_Decomposition
371  if (qbit_num > 0 ) {
372  if ( strcmp("limited", method_C)==0 or strcmp("LIMITED", method_C)==0) {
373  self->decomp = create_N_Qubit_Decomposition_adaptive( Umtx_mtx, qbit_num, level_limit, level_limit_min, topology_Cpp);
374  self->decomp_base = (N_Qubit_Decomposition_Base*)self->decomp;
375  }
376  else if ( strcmp("general", method_C)==0 or strcmp("GENERAL", method_C)==0) {
377  self->decomp_general = create_N_Qubit_Decomposition_adaptive_general( Umtx_mtx, qbit_num, level_limit, level_limit_min, topology_Cpp);
378  self->decomp_base = (N_Qubit_Decomposition_Base*)self->decomp_general;
379  }
380  else {
381  std::cout << "Wrong optmimization method. Falling back to limited." << std::endl;
382  self->decomp = create_N_Qubit_Decomposition_adaptive( Umtx_mtx, qbit_num, level_limit, level_limit_min, topology_Cpp );
383  self->decomp_base = (N_Qubit_Decomposition_Base*)self->decomp;
384  }
385  }
386  else {
387  std::cout << "The number of qubits should be given as a positive integer, " << qbit_num << " was given" << std::endl;
388  Py_XDECREF(method_string);
389  Py_XDECREF(method_string_unicode);
390  return -1;
391  }
392 
393 
394 
395  Py_XDECREF(method_string);
396  Py_XDECREF(method_string_unicode);
397 
398  return 0;
399 }

755 static PyObject *
756 qgd_N_Qubit_Decomposition_adaptive_Wrapper_get_Optimized_Parameters( qgd_N_Qubit_Decomposition_adaptive_Wrapper *self ) {

758  int parameter_num = self->decomp->get_parameter_num();
759 
760  matrix_base<double> parameters_mtx(1, parameter_num);
761  double* parameters = parameters_mtx.get_data();
762  self->decomp_base->get_optimized_parameters(parameters);
763 
764  // reversing the order
765  matrix_base<double> parameters_mtx_reversed(1, parameter_num);
766  double* parameters_reversed = parameters_mtx_reversed.get_data();
767  for (int idx=0; idx<parameter_num; idx++ ) {
768  parameters_reversed[idx] = parameters[parameter_num-1-idx];
769  }
770 
771  // convert to numpy array
772  parameters_mtx_reversed.set_owner(false);
773  PyObject * parameter_arr = matrix_real_to_numpy( parameters_mtx_reversed );
