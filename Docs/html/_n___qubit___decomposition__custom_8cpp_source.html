172 N_Qubit_Decomposition_custom::add_gate_layers() {
173 
174  release_gates();
175 
177  // add custom gate structure
178 
179  // the number of succeeding identical layers in the subdecomposition
180  int identical_blocks_loc;
181  try {
182  identical_blocks_loc = identical_blocks[qbit_num];
183  if (identical_blocks_loc==0) {
184  identical_blocks_loc = 1;
185  }
186  }
187  catch (...) {
188  identical_blocks_loc=1;
189  }
190 
191  // get the list of sub-blocks in the custom gate structure
192  std::vector<Gate*> gates = gate_structure->get_gates();
193 
194  for (std::vector<Gate*>::iterator gates_it = gates.begin(); gates_it != gates.end(); gates_it++ ) {
195 
196  // The current gate
197  Gate* gate = *gates_it;
198 
199  for (int idx=0; idx<identical_blocks_loc; idx++) {
200  if (gate->get_type() == CNOT_OPERATION ) {
201  CNOT* cnot_gate = static_cast<CNOT*>( gate );
202  add_gate_to_end( (Gate*)cnot_gate->clone() );
203  }
204  else if (gate->get_type() == CZ_OPERATION ) {
205  CZ* cz_gate = static_cast<CZ*>( gate );
206  add_gate_to_end( (Gate*)cz_gate->clone() );
207  }
208  else if (gate->get_type() == CH_OPERATION ) {
209  CH* ch_gate = static_cast<CH*>( gate );
210  add_gate_to_end( (Gate*)ch_gate->clone() );
211  }
212  else if (gate->get_type() == SYC_OPERATION ) {
213  SYC* syc_gate = static_cast<SYC*>( gate );
214  add_gate_to_end( (Gate*)syc_gate->clone() );
215  }
216  else if (gate->get_type() == GENERAL_OPERATION ) {
217  add_gate_to_end( gate->clone() );
218  }
219  else if (gate->get_type() == U3_OPERATION ) {
220  U3* u3_gate = static_cast<U3*>( gate );
221  add_gate_to_end( (Gate*)u3_gate->clone() );
222  }
223  else if (gate->get_type() == RX_OPERATION ) {
224  RX* rx_gate = static_cast<RX*>( gate );
225  add_gate_to_end( (Gate*)rx_gate->clone() );
226  }

229  add_gate_to_end( (Gate*)ry_gate->clone() );
230  }
231  else if (gate->get_type() == CRY_OPERATION ) {

234  }

242  }
