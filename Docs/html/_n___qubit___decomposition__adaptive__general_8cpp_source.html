358 bool are_two_qubits_independent( Gates_block* two_qubit_gate, Matrix_real& params ) {
359 
360  logging output;
361 
362  std::vector<int> involved_qubits = two_qubit_gate->get_involved_qubits();
363  if ( involved_qubits.size() > 2 ) {
364  std::stringstream sstream;
365  sstream << "N_Qubit_Decomposition_adaptive::are_two_qubits_independent: the givel block contains more than 2 qubits" << std::endl;
366  output.print(sstream, 0);
367  exit(-1);
368  }
369 
370  Matrix mtx_loc = two_qubit_gate->get_matrix( params );
371  QGD_Complex16* mtx_data = mtx_loc.get_data();
372  int dim = mtx_loc.rows;
373 
374  // reorder the matrix elements
375  Matrix mtx_new( dim, dim );
376  QGD_Complex16* mtx_new_data = mtx_new.get_data();
377 
378  int qbit = involved_qubits[0];
379  size_t dim_loc = (size_t)Power_of_2(qbit);
380  size_t dim_loc2 = (size_t)dim_loc*2;
381  int dim_over_2 = dim/2;
382 
383  for ( size_t row_idx=0; row_idx<dim_over_2; row_idx++ ) {
384 
385  int row_tmp = row_idx % dim_loc;
386  int row_tmp2 = (row_idx-row_tmp)/dim_loc;
387 
388  for ( size_t col_idx=0; col_idx<dim_over_2; col_idx+=dim_loc ) {
389 
390  memcpy( mtx_new_data + row_idx*dim + col_idx, mtx_data + (row_tmp2*dim_loc2+row_tmp)*dim + (col_idx*2), dim_loc*sizeof(QGD_Complex16) );
391  memcpy( mtx_new_data + row_idx*dim + col_idx + dim_over_2, mtx_data + (row_tmp2*dim_loc2+row_tmp)*dim + (col_idx*2 + dim_loc), dim_loc*sizeof(QGD_Complex16) );
392  memcpy( mtx_new_data + (row_idx + dim_over_2)*dim + col_idx, mtx_data + (row_tmp2*dim_loc2+row_tmp+dim_loc)*dim + (col_idx*2), dim_loc*sizeof(QGD_Complex16) );
393  memcpy( mtx_new_data + (row_idx + dim_over_2)*dim + col_idx + dim_over_2, mtx_data + (row_tmp2*dim_loc2+row_tmp+dim_loc)*dim + (col_idx*2 + dim_loc), dim_loc*sizeof(QGD_Complex16) );
394 
395 
396  }
397  }
398 
399  double cost_fnc = get_submatrix_cost_function(mtx_new);
400 //std::cout << cost_fnc << std::endl;
401  if (cost_fnc < 1e-2 ) return true;
402 
403  return false;
