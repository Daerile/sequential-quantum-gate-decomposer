198 void Composite::reorder_qubits( std::vector<int> qbit_list ) {
199 
200  // check the number of qubits
201  if ((int)qbit_list.size() != qbit_num ) {
202  std::stringstream sstream;
203  sstream << "Wrong number of qubits" << std::endl;
204  print(sstream, 0);
205  exit(-1);
206  }
207 
208 
209  int control_qbit_new = control_qbit;
210  int target_qbit_new = target_qbit;
211 
212  // setting the new value for the target qubit
213  for (int idx=0; idx<qbit_num; idx++) {
214  if (target_qbit == qbit_list[idx]) {
215  target_qbit_new = qbit_num-1-idx;
216  }
217  if (control_qbit == qbit_list[idx]) {
218  control_qbit_new = qbit_num-1-idx;
219  }
220  }
221 
222  control_qbit = control_qbit_new;
223  target_qbit = target_qbit_new;
224 }
