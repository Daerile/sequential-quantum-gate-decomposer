265 Matrix Decomposition_Base::get_finalizing_gates( Matrix& mtx, Gates_block* finalizing_gates, double* finalizing_parameters ) {
266 
267 
268  int parameter_idx = finalizing_parameter_num-1;
269 
270  Matrix mtx_tmp = mtx.copy();
271 
272 
273  double Theta, Lambda, Phi;
274  for (int target_qbit=0; target_qbit<qbit_num; target_qbit++ ) {
275 
276  // get the base indices of the taget qubit states |0>, where all other qubits are in state |0>
277  int state_0 = 0;
278 
279  // get the base indices of the taget qubit states |1>, where all other qubits are in state |0>
280  int state_1 = Power_of_2(target_qbit);
281 
282  // finalize the 2x2 submatrix with z-y-z rotation
283  QGD_Complex16 element00 = mtx[state_0*matrix_size+state_0];
284  QGD_Complex16 element01 = mtx[state_0*matrix_size+state_1];
285  QGD_Complex16 element10 = mtx[state_1*matrix_size+state_0];
286  QGD_Complex16 element11 = mtx[state_1*matrix_size+state_1];
287 
288  // finalize the 2x2 submatrix with z-y-z rotation
289  double cos_theta_2 = sqrt(element00.real*element00.real + element00.imag*element00.imag)/sqrt(element00.real*element00.real + element00.imag*element00.imag + element01.real*element01.real + element01.imag*element01.imag);
290  Theta = 2*acos( cos_theta_2 );
291 
292  if ( sqrt(element00.real*element00.real + element00.imag*element00.imag) < 1e-7 ) {
293  Phi = atan2(element10.imag, element10.real); //np.angle( submatrix[1,0] )
294  Lambda = atan2(-element01.imag, -element01.real); //np.angle( -submatrix[0,1] )
295  }
296  else if ( sqrt(element10.real*element10.real + element10.imag*element10.imag) < 1e-7 ) {
297  Phi = 0;
298  Lambda = atan2(element11.imag*element00.real - element11.real*element00.imag, element11.real*element00.real + element11.imag*element00.imag); //np.angle( element11*np.conj(element00))
299  }
300  else {
301  Phi = atan2(element10.imag*element00.real - element10.real*element00.imag, element10.real*element00.real + element10.imag*element00.imag); //np.angle( element10*np.conj(element00))
302  Lambda = atan2(-element01.imag*element00.real + element01.real*element00.imag, -element01.real*element00.real - element01.imag*element00.imag); //np.angle( -element01*np.conj(element00))
303  }
304 
305  double parameters_loc[3];
306  parameters_loc[0] = Theta;
307  parameters_loc[1] = M_PI-Lambda;
308  parameters_loc[2] = M_PI-Phi;
309 
310  U3* u3_loc = new U3( qbit_num, target_qbit, true, true, true);
311 
312  // adding the new gate to the list of finalizing gates
313  finalizing_parameters[parameter_idx] = M_PI-Phi; //np.concatenate((parameters_loc, finalizing_parameters))
314  parameter_idx--;
315  finalizing_parameters[parameter_idx] = M_PI-Lambda; //np.concatenate((parameters_loc, finalizing_parameters))
316  parameter_idx--;
317  finalizing_parameters[parameter_idx] = Theta; //np.concatenate((parameters_loc, finalizing_parameters))
318  parameter_idx--;
319 
320  finalizing_gates->add_gate( u3_loc );
321  // get the new matrix
322 
323 
324  Matrix u3_mtx = u3_loc->get_matrix(parameters_loc);
325 
326  Matrix tmp2 = apply_gate( u3_mtx, mtx_tmp);
327  mtx_tmp = tmp2;
328 
329  }
330 
331 
332  return mtx_tmp;

345 void Decomposition_Base::solve_optimization_problem( double* solution_guess, int solution_guess_num ) {
346 
347 
348  if ( gates.size() == 0 ) {
349  return;
350  }
351 
352  // array containing minimums to check convergence of the solution
353  int min_vec_num = 20;
354  double minimum_vec[min_vec_num];
355  for ( int idx=0; idx<min_vec_num; idx++) {
356  minimum_vec[idx] = 0;
357  }
358 
359  // setting the initial value for the current minimum
360  current_minimum = 1e8;
361 
362  // store the gates
363  std::vector<Gate*> gates_loc = gates;
364 
365 
366 
367  // store the number of parameters
368  int parameter_num_loc = parameter_num;
369 
370  // store the initial unitary to be decomposed
371  Matrix Umtx_loc = Umtx; // copy??
372 
373  // storing the initial computational parameters
374  int optimization_block_loc = optimization_block;
375 
376  // initialize random seed:
377  srand (time(NULL));
378 
379  // the array storing the optimized parameters
380  gsl_vector* optimized_parameters_gsl = gsl_vector_alloc (parameter_num_loc);
381 
382  // preparing solution guess for the iterations
383  if ( initial_guess == ZEROS ) {
384  for(unsigned int idx = 0; idx < parameter_num-solution_guess_num; idx++) {
385  optimized_parameters_gsl->data[idx] = 0;
386  }
387  }
388  else if ( initial_guess == RANDOM ) {
389  for(unsigned int idx = 0; idx < parameter_num-solution_guess_num; idx++) {
390  optimized_parameters_gsl->data[idx] = (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
391  }
392  }
393  else if ( initial_guess == CLOSE_TO_ZERO ) {
394  for(unsigned int idx = 0; idx < parameter_num-solution_guess_num; idx++) {
395  optimized_parameters_gsl->data[idx] = (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
396  }
397  }
398  else {
399  printf("bad value for initial guess\n");
400  exit(-1);
401  }
402 
403  if ( solution_guess_num > 0) {
404  memcpy(optimized_parameters_gsl->data + parameter_num-solution_guess_num, solution_guess, solution_guess_num*sizeof(double));
405  }
406 
407  // starting number of gate block applied prior to the optimalized gate blocks
408  int pre_gate_parameter_num = 0;
409 
410  // defining temporary variables for iteration cycles
411  int block_idx_end;
412  unsigned int block_idx_start = gates.size();
413  gates.clear();
414  int block_parameter_num;
415  Matrix gates_mtx_pre;
416  Gate* fixed_gate_post = new Gate( qbit_num );
417  std::vector<Matrix, tbb::cache_aligned_allocator<Matrix>> gates_mtxs_post;
418 
419  // the identity matrix used in the calculations
420  Matrix Identity = create_identity( matrix_size );
421 
422 
423  gsl_vector *solution_guess_gsl = NULL;
424 
425 
426  //measure the time for the decomposition
427  clock_t start_time = time(NULL);
428 
430  // Start the iterations
431  int iter_idx;
432  for ( iter_idx=0; iter_idx<max_iterations+1; iter_idx++) {
433 
434  //determine the range of blocks to be optimalized togedther
435  block_idx_end = block_idx_start - optimization_block;
436  if (block_idx_end < 0) {
437  block_idx_end = 0;
438  }
439 
440  // determine the number of free parameters to be optimized
441  block_parameter_num = 0;
442  for ( int block_idx=block_idx_start-1; block_idx>=block_idx_end; block_idx--) {
443  block_parameter_num = block_parameter_num + gates_loc[block_idx]->get_parameter_num();
444  }
445 
446  // ***** get the fixed gates applied before the optimized gates *****
447  if (block_idx_start < gates_loc.size() ) {
448  std::vector<Gate*>::iterator fixed_gates_pre_it = gates.begin() + 1;
449  Matrix tmp = get_transformed_matrix(optimized_parameters, fixed_gates_pre_it, gates.size()-1, gates_mtx_pre);
450  gates_mtx_pre = tmp; // copy?
451  }
452  else {
453  gates_mtx_pre = Identity.copy();
454  }
455 
456  // Transform the initial unitary upon the fixed pre-optimization gates
457  Umtx = apply_gate(gates_mtx_pre, Umtx_loc);
458 
459  // clear the gate list used in the previous iterations
460  gates.clear();
461 
462  if (optimized_parameters != NULL ) {
463  qgd_free( optimized_parameters );
464  optimized_parameters = NULL;
465  }
466 
467 
468  // ***** get the fixed gates applied after the optimized gates *****
469  // create a list of post gates matrices
470  if (block_idx_start == gates_loc.size() ) {
471  // matrix of the fixed gates aplied after the gates to be varied
472  double* fixed_parameters_post = optimized_parameters_gsl->data;
473  std::vector<Gate*>::iterator fixed_gates_post_it = gates_loc.begin();
474 
475  gates_mtxs_post = get_gate_products(fixed_parameters_post, fixed_gates_post_it, block_idx_end);
476  }
477 
478  // Create a general gate describing the cumulative effect of gates following the optimized gates
479  if (block_idx_end > 0) {
480  fixed_gate_post->set_matrix( gates_mtxs_post[block_idx_end-1] );
481  }
482  else {
483  // release gate products
484  gates_mtxs_post.clear();
485  fixed_gate_post->set_matrix( Identity );
486  }
487 
488  // create a list of gates for the optimization process
489  gates.push_back( fixed_gate_post );
490  for ( unsigned int idx=block_idx_end; idx<block_idx_start; idx++ ) {
491  gates.push_back( gates_loc[idx] );
492  }
493 
494 
495  // constructing solution guess for the optimization
496  parameter_num = block_parameter_num;
497  if ( solution_guess_gsl == NULL ) {
498  solution_guess_gsl = gsl_vector_alloc (parameter_num);
499  }
500  else if ( parameter_num != solution_guess_gsl->size ) {
501  gsl_vector_free(solution_guess_gsl);
502  solution_guess_gsl = gsl_vector_alloc (parameter_num);
503  }
504  memcpy( solution_guess_gsl->data, optimized_parameters_gsl->data+parameter_num_loc - pre_gate_parameter_num - block_parameter_num, parameter_num*sizeof(double) );
505 
506  // solve the optimization problem of the block
507  solve_layer_optimization_problem( parameter_num, solution_guess_gsl );
508 
509  // add the current minimum to the array of minimums and calculate the mean
510  double minvec_mean = 0;
511  for (int idx=min_vec_num-1; idx>0; idx--) {
512  minimum_vec[idx] = minimum_vec[idx-1];
513  minvec_mean = minvec_mean + minimum_vec[idx-1];
514  }
515  minimum_vec[0] = current_minimum;
516  minvec_mean = minvec_mean + current_minimum;
517  minvec_mean = minvec_mean/min_vec_num;
518 
519  // store the obtained optimalized parameters for the block
520  memcpy( optimized_parameters_gsl->data+parameter_num_loc - pre_gate_parameter_num-block_parameter_num, optimized_parameters, parameter_num*sizeof(double) );
521 
522  if (block_idx_end == 0) {
523  block_idx_start = gates_loc.size();
524  pre_gate_parameter_num = 0;
525  }
526  else {
527  block_idx_start = block_idx_start - optimization_block;
528  pre_gate_parameter_num = pre_gate_parameter_num + block_parameter_num;
529  }
530 
531 
532  // optimization result is displayed in each 500th iteration
533  if (iter_idx % 500 == 0) {
534  if (verbose) {
535  printf("The minimum with %d layers after %d iterations is %e calculated in %f seconds\n", layer_num, iter_idx, current_minimum, float(time(NULL) - start_time));
536  fflush(stdout);
537  }
538  start_time = time(NULL);
539  }
540 
541 
542  // calculate the variance of the last 10 minimums
543  double minvec_std = sqrt(gsl_stats_variance_m( minimum_vec, 1, min_vec_num, minvec_mean));
544 
545  // conditions to break the iteration cycles
546  if (std::abs(minvec_std/minimum_vec[min_vec_num-1]) < optimization_tolerance ) {
547  if (verbose) {
548  printf("The iterations converged to minimum %e after %d iterations with %d layers\n", current_minimum, iter_idx, layer_num );
549  fflush(stdout);
550  }
551  break;
552  }
553  else if (check_optimization_solution()) {
554  if (verbose) {
555  printf("The minimum with %d layers after %d iterations is %e\n", layer_num, iter_idx, current_minimum);
556  }
557  break;
558  }
559 
560 
561  }
562 
563 
564  if (iter_idx == max_iterations ) {
565  if (verbose) {
566  printf("Reached maximal number of iterations\n\n");
567  }
568  }
569 
570  // restoring the parameters to originals
571  optimization_block = optimization_block_loc;
572 
573  // store the result of the optimization
574  gates.clear();
575  gates = gates_loc;
576 
577  parameter_num = parameter_num_loc;
578  if (optimized_parameters != NULL ) {
579  qgd_free( optimized_parameters );
580  }
581 
582  optimized_parameters = (double*)qgd_calloc(parameter_num,sizeof(double), CACHELINE);
583  memcpy( optimized_parameters, optimized_parameters_gsl->data, parameter_num*sizeof(double) );
584 
585 
586  // free unnecessary resources
587  gsl_vector_free(optimized_parameters_gsl);
588  gsl_vector_free(solution_guess_gsl);
589  optimized_parameters_gsl = NULL;
590  solution_guess_gsl = NULL;
591 
592  delete(fixed_gate_post);
593 
594  // restore the original unitary
595  Umtx = Umtx_loc; // copy?
596 
597 
598 }

696 Decomposition_Base::get_transformed_matrix( const double* parameters, std::vector<Gate*>::iterator gates_it, int num_of_gates, Matrix& initial_matrix ) {
697 
698  if (num_of_gates==0) {
699  return initial_matrix.copy();
700  }
701 
702  // The matrix of the transformed matrix
703  Matrix Gate_product;
704 
705  // The matrix to be returned
706  Matrix ret_matrix;
707 
708  for (int idx=0; idx<num_of_gates; idx++) {
709 
710  // The current gate
711  Gate* gate = *gates_it;
712 
713  // The matrix of the current gate
714  Matrix gate_mtx;
715 
716  if (gate->get_type() == CNOT_OPERATION ) {
717  CNOT* cnot_gate = static_cast<CNOT*>( gate );
718  gate_mtx = cnot_gate->get_matrix();
719  }
720  else if (gate->get_type() == CZ_OPERATION ) {
721  CZ* cz_gate = static_cast<CZ*>( gate );
722  gate_mtx = cz_gate->get_matrix();
723  }
724  else if (gate->get_type() == CH_OPERATION ) {
725  CH* ch_gate = static_cast<CH*>( gate );
726  gate_mtx = ch_gate->get_matrix();
727  }
728  else if (gate->get_type() == GENERAL_OPERATION ) {
729  gate_mtx = gate->get_matrix();
730  }
731  else if (gate->get_type() == U3_OPERATION ) {
732  U3* u3_gate = static_cast<U3*>( gate );
733  int parameters_num = u3_gate->get_parameter_num();
734  gate_mtx = u3_gate->get_matrix( parameters );
735  parameters = parameters + parameters_num;
736  }
737  else if (gate->get_type() == BLOCK_OPERATION ) {
738  Gates_block* block_gate = static_cast<Gates_block*>( gate );
739  int parameters_num = block_gate->get_parameter_num();
740  gate_mtx = block_gate->get_matrix( parameters );
741  parameters = parameters + parameters_num;

1008  vartheta = std::fmod( parameters[ parameter_idx ], 4*M_PI);
1009  varphi = std::fmod( parameters[ parameter_idx+1 ], 2*M_PI);
1010  varlambda = 0;
1011  parameter_idx = parameter_idx + 2;
1012  }
