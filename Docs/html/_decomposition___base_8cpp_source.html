222  for (int target_qbit=0; target_qbit<qbit_num; target_qbit++ ) {
223 
224  // get the base indices of the taget qubit states |0>, where all other qubits are in state |0>
225  int state_0 = 0;
226 
227  // get the base indices of the taget qubit states |1>, where all other qubits are in state |0>
228  int state_1 = Power_of_2(target_qbit);
229 
230  // finalize the 2x2 submatrix with z-y-z rotation
231  QGD_Complex16 element00 = mtx[state_0*matrix_size+state_0];
232  QGD_Complex16 element01 = mtx[state_0*matrix_size+state_1];
233  QGD_Complex16 element10 = mtx[state_1*matrix_size+state_0];
234  QGD_Complex16 element11 = mtx[state_1*matrix_size+state_1];
235 
236  // finalize the 2x2 submatrix with z-y-z rotation
237  double cos_theta_2 = sqrt(element00.real*element00.real + element00.imag*element00.imag)/sqrt(element00.real*element00.real + element00.imag*element00.imag + element01.real*element01.real + element01.imag*element01.imag);
238  Theta = 2*acos( cos_theta_2 );
239

241  Phi = atan2(element10.imag, element10.real); //np.angle( submatrix[1,0] )
242  Lambda = atan2(-element01.imag, -element01.real); //np.angle( -submatrix[0,1] )
243  }
244  else if ( sqrt(element10.real*element10.real + element10.imag*element10.imag) < 1e-7 ) {
245  Phi = 0;
246  Lambda = atan2(element11.imag*element00.real - element11.real*element00.imag, element11.real*element00.real + element11.imag*element00.imag); //np.angle( element11*np.conj(element00))
247  }

249  Phi = atan2(element10.imag*element00.real - element10.real*element00.imag, element10.real*element00.real + element10.imag*element00.imag); //np.angle( element10*np.conj(element00))
250  Lambda = atan2(-element01.imag*element00.real + element01.real*element00.imag, -element01.real*element00.real - element01.imag*element00.imag); //np.angle( -element01*np.conj(element00))
251  }

259

273 
274 
275  memset(u3_mtx, 0, matrix_size*matrix_size*sizeof(QGD_Complex16) );

279 //print_mtx(u3_mtx, matrix_size, matrix_size );

315 void Decomposition_Base::solve_optimization_problem( double* solution_guess, int solution_guess_num ) {
316 
317  if ( operations.size() == 0 ) {
318  return;
319  }
320 
321 
322  // array containing minimums to check convergence of the solution
323  int min_vec_num = 20;
324  double minimum_vec[min_vec_num];
325  for ( int idx=0; idx<min_vec_num; idx++) {
326  minimum_vec[idx] = 0;
327  }
328 
329  // setting the initial value for the current minimum
330  current_minimum = 1e8;
331 
332  // store the operations
333  std::vector<Operation*> operations_loc = operations;
334 
335 
336 
337  // store the number of parameters
338  int parameter_num_loc = parameter_num;
339 
340  // store the initial unitary to be decomposed
341  QGD_Complex16* Umtx_loc = (QGD_Complex16*)qgd_calloc(matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
342  memcpy(Umtx_loc, Umtx, matrix_size*matrix_size*sizeof(QGD_Complex16) );
343 
344  // storing the initial computational parameters
345  int optimization_block_loc = optimization_block;
346 
347  // initialize random seed:
348  srand (time(NULL));
349 
350  // the array storing the optimized parameters
351  gsl_vector* optimized_parameters_gsl = gsl_vector_alloc (parameter_num_loc);
352 
353  // preparing solution guess for the iterations
354  if ( initial_guess == ZEROS ) {
355  for(unsigned int idx = 0; idx < parameter_num-solution_guess_num; idx++) {
356  optimized_parameters_gsl->data[idx] = 0;
357  }
358  }
359  else if ( initial_guess == RANDOM ) {
360  for(unsigned int idx = 0; idx < parameter_num-solution_guess_num; idx++) {
361  optimized_parameters_gsl->data[idx] = (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
362  }
363  }
364  else if ( initial_guess == CLOSE_TO_ZERO ) {
365  for(unsigned int idx = 0; idx < parameter_num-solution_guess_num; idx++) {
366  optimized_parameters_gsl->data[idx] = (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
367  }
368  }
369  else {
370  printf("bad value for initial guess\n");
371  exit(-1);
372  }
373 
374  if ( solution_guess_num > 0) {
375  memcpy(optimized_parameters_gsl->data + parameter_num-solution_guess_num, solution_guess, solution_guess_num*sizeof(double));
376  }
377 
378  // starting number of operation block applied prior to the optimalized operation blocks
379  int pre_operation_parameter_num = 0;
380 
381  // defining temporary variables for iteration cycles
382  int block_idx_end;
383  unsigned int block_idx_start = operations.size();
384  operations.clear();
385  int block_parameter_num;
386  QGD_Complex16* operations_mtx_pre = (QGD_Complex16*)qgd_calloc(matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
387  QGD_Complex16* tmp = NULL;
388  Operation* fixed_operation_post = new Operation( qbit_num );
389  std::vector<QGD_Complex16*> operations_mtxs_post;
390 
391  // the identity matrix used in the calcualtions
392  QGD_Complex16* Identity = (QGD_Complex16*)qgd_calloc(matrix_size*matrix_size, sizeof(QGD_Complex16), 64);
393  create_identity( Identity, matrix_size );
394 
395  gsl_vector *solution_guess_gsl = NULL;
396 
397 
398  //measure the time for the decompositin
399  clock_t start_time = time(NULL);
400 
401  int iter_idx;
402  for ( iter_idx=0; iter_idx<max_iterations+1; iter_idx++) {
403 
404  //determine the range of blocks to be optimalized togedther
405  block_idx_end = block_idx_start - optimization_block;
406  if (block_idx_end < 0) {
407  block_idx_end = 0;
408  }
409 
410  // determine the number of free parameters to be optimized
411  block_parameter_num = 0;
412  for ( int block_idx=block_idx_start-1; block_idx>=block_idx_end; block_idx--) { //for block_idx in range(block_idx_start-1,block_idx_end-1,-1):
413  block_parameter_num = block_parameter_num + operations_loc[block_idx]->get_parameter_num();
414  }
415 
416  // ***** get the fixed operations applied before the optimized operations *****
417  if (block_idx_start < operations_loc.size() ) { //if block_idx_start < len(operations):
418  std::vector<Operation*>::iterator fixed_operations_pre_it = operations.begin() + 1;
419  tmp = get_transformed_matrix(optimized_parameters, fixed_operations_pre_it, operations.size()-1, operations_mtx_pre);
420  memcpy( operations_mtx_pre, tmp, matrix_size*matrix_size*sizeof(QGD_Complex16) );
421  qgd_free(tmp);
422  tmp = NULL;
423  //QGD_Complex16* operations_mtx_pre_tmp = get_transformed_matrix(optimized_parameters, fixed_operations_pre_it, operations.size()-1, operations_mtx_pre );
424  //qgd_free( operations_mtx_pre );
425  //operations_mtx_pre = operations_mtx_pre_tmp;
426  }
427  else {
428  memcpy( operations_mtx_pre, Identity, matrix_size*matrix_size*sizeof(QGD_Complex16) );
429  }
430 
431 //print_mtx(operations_mtx_pre, matrix_size, matrix_size );
433 //operations_mtx_pre = Identity;
435 
436  // Transform the initial unitary upon the fixed pre-optimization operations
437  apply_operation(operations_mtx_pre, Umtx_loc, Umtx);
438 
439 
440  // clear the operation list used in the previous iterations
441  operations.clear();
442 
443  if (optimized_parameters != NULL ) {
444  qgd_free( optimized_parameters );
445  optimized_parameters = NULL;
446  }
447 
448 
449  // ***** get the fixed operations applied after the optimized operations *****
450  // create a list of post operations matrices
451  if (block_idx_start == operations_loc.size() ) {
452  // matrix of the fixed operations aplied after the operations to be varied
453  double* fixed_parameters_post = optimized_parameters_gsl->data;
454  std::vector<Operation*>::iterator fixed_operations_post_it = operations_loc.begin();
455  operations_mtxs_post = get_operation_products(fixed_parameters_post, fixed_operations_post_it, block_idx_end);
456  }
457 
458  // Create a general operation describing the cumulative effect of gates following the optimized operations
459  if (block_idx_end > 0) {
460  fixed_operation_post->set_matrix( operations_mtxs_post[block_idx_end-1] );
461 //printf("operations_mtxs_post with prefixed:\n");
462 //print_mtx(operations_mtxs_post[block_idx_end-1], matrix_size, matrix_size);
463  }
464  else {
465  // release operation products
466  for (std::vector<QGD_Complex16*>::iterator mtxs_it=operations_mtxs_post.begin(); mtxs_it != operations_mtxs_post.end(); mtxs_it++ ) {
467  qgd_free( *mtxs_it );
468  *mtxs_it = NULL;
469  }
470  operations_mtxs_post.clear();
471  fixed_operation_post->set_matrix( Identity );
472  }
473 
474 
476 //fixed_operation_post->set_matrix( Identity );
478 
479  // create a list of operations for the optimization process
480  operations.push_back( fixed_operation_post );
481  for ( unsigned int idx=block_idx_end; idx<block_idx_start; idx++ ) {
482  operations.push_back( operations_loc[idx] );
483  }
484 
485 
486  // constructing solution guess for the optimization
487  parameter_num = block_parameter_num;
488  if ( solution_guess_gsl == NULL ) {
489  solution_guess_gsl = gsl_vector_alloc (parameter_num);
490  }
491  else if ( parameter_num != solution_guess_gsl->size ) {
492  gsl_vector_free(solution_guess_gsl);
493  solution_guess_gsl = gsl_vector_alloc (parameter_num);
494  }
495  memcpy( solution_guess_gsl->data, optimized_parameters_gsl->data+parameter_num_loc - pre_operation_parameter_num - block_parameter_num, parameter_num*sizeof(double) );
496 
497  // solve the optimization problem of the block
498  solve_layer_optimization_problem( parameter_num, solution_guess_gsl );
499 
500  // add the current minimum to the array of minimums and calculate the mean
501  double minvec_mean = 0;
502  for (int idx=min_vec_num-1; idx>0; idx--) {
503  minimum_vec[idx] = minimum_vec[idx-1];
504  minvec_mean = minvec_mean + minimum_vec[idx-1];
505  }
506  minimum_vec[0] = current_minimum;
507  minvec_mean = minvec_mean + current_minimum;
508  minvec_mean = minvec_mean/min_vec_num;
509 
510  // store the obtained optimalized parameters for the block
511  memcpy( optimized_parameters_gsl->data+parameter_num_loc - pre_operation_parameter_num-block_parameter_num, optimized_parameters, parameter_num*sizeof(double) );
512 
513  if (block_idx_end == 0) {
514  block_idx_start = operations_loc.size();
515  pre_operation_parameter_num = 0;
516  }
517  else {
518  block_idx_start = block_idx_start - optimization_block;
519  pre_operation_parameter_num = pre_operation_parameter_num + block_parameter_num;
520  }
521 
522 
523  // optimization result is displayed in each 500th iteration
524  if (iter_idx % 500 == 0) {
525  if (verbose) {
526  printf("The minimum with %d layers after %d iterations is %e calculated in %f seconds\n", layer_num, iter_idx, current_minimum, float(time(NULL) - start_time));
527  fflush(stdout);
528  }
529  start_time = time(NULL);
530  }
531 
532 
533  // calculate the variance of the last 10 minimums
534  double minvec_std = sqrt(gsl_stats_variance_m( minimum_vec, 1, min_vec_num, minvec_mean));
535 
536  // conditions to break the iteration cycles
537  if (abs(minvec_std/minimum_vec[min_vec_num-1]) < optimization_tolerance ) {
538  if (verbose) {
539  printf("The iterations converged to minimum %e after %d iterations with %d layers\n", current_minimum, iter_idx, layer_num );
540  fflush(stdout);
541  }
542  break;
543  }
544  else if (check_optimization_solution()) {
545  if (verbose) {
546  printf("The minimum with %d layers after %d iterations is %e\n", layer_num, iter_idx, current_minimum);
547  }
548  break;
549  }
550 
551 
552  }
553 
554 
555  if (iter_idx == max_iterations ) {
556  if (verbose) {
557  printf("Reached maximal number of iterations\n\n");
558  }
559  }
560 
561  // restoring the parameters to originals
562  optimization_block = optimization_block_loc;
563 
564  // store the result of the optimization
565  operations.clear();
566  operations = operations_loc;
567 
568  parameter_num = parameter_num_loc;
569  if (optimized_parameters != NULL ) {
570  qgd_free( optimized_parameters );
571  optimized_parameters = (double*)qgd_calloc(parameter_num,sizeof(double), 64);
572  }
573 
574  memcpy( optimized_parameters, optimized_parameters_gsl->data, parameter_num*sizeof(double) );
575 
576 
577  // free unnecessary resources
578  gsl_vector_free(optimized_parameters_gsl);
579  gsl_vector_free(solution_guess_gsl);
580  optimized_parameters_gsl = NULL;
581  solution_guess_gsl = NULL;
582 
583  QGD_Complex16* operations_mtx_post = fixed_operation_post->matrix();
584 
585  // Identity can be freed if operations_mtx_post and operations_mtx_pre are not equal to identity
586  if ( operations_mtx_post!=Identity ) {
587  qgd_free(Identity);
588  }
589  Identity = NULL;
590 
591 
592  // release preoperation matrix
593  qgd_free( operations_mtx_pre );
594  operations_mtx_pre = NULL;
595 
596  // release post operation products
597  for (std::vector<QGD_Complex16*>::iterator mtxs_it=operations_mtxs_post.begin(); mtxs_it != operations_mtxs_post.end(); mtxs_it++ ) {
598  qgd_free( *mtxs_it );
599  *mtxs_it = NULL;
600  }

602 
603

608

611  Umtx_loc = NULL;
612

940 std::vector<Operation*> Decomposition_Base::prepare_operations_to_export( std::vector<Operation*> ops, const double* parameters ) {
941 
942  std::vector<Operation*> ops_ret;
943  int parameter_idx = 0;
944 
945 
946  for(std::vector<Operation*>::iterator it = ops.begin(); it != ops.end(); it++) {
947 
948  Operation* operation = *it;
949 
950  if (operation->get_type() == CNOT_OPERATION) {
951  ops_ret.push_back( operation );
952  }
953  else if (operation->get_type() == U3_OPERATION) {
954 
955  // definig the U3 parameters
956  double vartheta;
957  double varphi;
958  double varlambda;
959 
960  // get the inverse parameters of the U3 rotation
961 
962  U3* u3_operation = static_cast<U3*>(operation);
963 
964  if ((u3_operation->get_parameter_num() == 1) && u3_operation->is_theta_parameter()) {
965  vartheta = std::fmod( parameters[parameter_idx], 4*M_PI);
966  varphi = 0;
967  varlambda =0;
968  parameter_idx = parameter_idx + 1;
969 
970  }
971  else if ((u3_operation->get_parameter_num() == 1) && u3_operation->is_phi_parameter()) {
972  vartheta = 0;
973  varphi = std::fmod( parameters[ parameter_idx ], 2*M_PI);
974  varlambda =0;
975  parameter_idx = parameter_idx + 1;
976  }
977  else if ((u3_operation->get_parameter_num() == 1) && u3_operation->is_lambda_parameter()) {
978  vartheta = 0;
979  varphi = 0;
980  varlambda = std::fmod( parameters[ parameter_idx ], 2*M_PI);
981  parameter_idx = parameter_idx + 1;
982  }
983  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_theta_parameter() && u3_operation->is_phi_parameter() ) {
984  vartheta = std::fmod( parameters[ parameter_idx ], 4*M_PI);
985  varphi = std::fmod( parameters[ parameter_idx+1 ], 2*M_PI);
986  varlambda = 0;
987  parameter_idx = parameter_idx + 2;
988  }
989  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_theta_parameter() && u3_operation->is_lambda_parameter() ) {
990  vartheta = std::fmod( parameters[ parameter_idx ], 4*M_PI);
991  varphi = 0;
992  varlambda = std::fmod( parameters[ parameter_idx+1 ], 2*M_PI);
993  parameter_idx = parameter_idx + 2;
994  }
995  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_phi_parameter() && u3_operation->is_lambda_parameter() ) {
996  vartheta = 0;
997  varphi = std::fmod( parameters[ parameter_idx], 2*M_PI);
998  varlambda = std::fmod( parameters[ parameter_idx+1 ], 2*M_PI);
999  parameter_idx = parameter_idx + 2;
1000  }
1001  else if ((u3_operation->get_parameter_num() == 3)) {
1002  vartheta = std::fmod( parameters[ parameter_idx ], 4*M_PI);
1003  varphi = std::fmod( parameters[ parameter_idx+1 ], 2*M_PI);
1004  varlambda = std::fmod( parameters[ parameter_idx+2 ], 2*M_PI);
1005  parameter_idx = parameter_idx + 3;
1006  }
1007  else {
1008  printf("wrong parameters in U3 class\n");
1009  exit(-1);
1010  }
1011 
1012  u3_operation->set_optimized_parameters( vartheta, varphi, varlambda );
1013  ops_ret.push_back( static_cast<Operation*>(u3_operation) );
1014 
1015 
1016  }
1017  else if (operation->get_type() == BLOCK_OPERATION) {
1018  Operation_block* block_operation = static_cast<Operation_block*>(operation);
1019  const double* parameters_layer = parameters + parameter_idx;
1020 
1021  std::vector<Operation*> ops_loc = prepare_operations_to_export(block_operation, parameters_layer);
1022  parameter_idx = parameter_idx + block_operation->get_parameter_num();
1023 
1024  ops_ret.insert( ops_ret.end(), ops_loc.begin(), ops_loc.end() );
1025  }
1026 
1027  }
