84 void functor_cost_fnc::operator()( int row_idx ) const {
85 
86  if ( row_idx > partial_cost_fnc_num ) {
87  printf("Error: row idx should be less than the number of roes in the matrix\n");
88  exit(-1);
89  }
90 
91  // getting the corner element
92  QGD_Complex16 corner_element = matrix[0];
93 
94 
95  // Calculate the |x|^2 value of the elements of the matrix and summing them up to calculate the partial cost function
96  double partial_cost_function = 0;
97  int idx_offset = row_idx*matrix_size;
98  int idx_max = idx_offset + row_idx;
99  for ( int idx=idx_offset; idx<idx_max; idx++ ) {
100  partial_cost_function = partial_cost_function + matrix[idx].real*matrix[idx].real + matrix[idx].imag*matrix[idx].imag;
101  }
102 
103  int diag_element_idx = row_idx*matrix_size + row_idx;
104  double diag_real = matrix[diag_element_idx].real - corner_element.real;
105  double diag_imag = matrix[diag_element_idx].imag - corner_element.imag;
106  partial_cost_function = partial_cost_function + diag_real*diag_real + diag_imag*diag_imag;
107 
108 
109  idx_offset = idx_max + 1;
110  idx_max = row_idx*matrix_size + matrix_size;
111  for ( int idx=idx_offset; idx<idx_max; idx++ ) {
112  partial_cost_function = partial_cost_function + matrix[idx].real*matrix[idx].real + matrix[idx].imag*matrix[idx].imag;
113  }
114 
115  // storing the calculated partial cost function
116  partial_cost_functions[row_idx] = partial_cost_function;
117 }
