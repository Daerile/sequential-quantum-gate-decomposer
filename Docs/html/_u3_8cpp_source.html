461 int U3::calc_one_qubit_u3(double Theta, double Phi, double Lambda ) {
462 
463  double cos_theta = cos(Theta/2);
464  double sin_theta = sin(Theta/2);
465 
466  // the 1,1 element
467  u3_1qbit[0].real = cos_theta;
468  u3_1qbit[0].imag = 0;
469  // the 1,2 element
470  u3_1qbit[1].real = -cos(Lambda)*sin_theta;
471  u3_1qbit[1].imag = -sin(Lambda)*sin_theta;
472  // the 2,1 element
473  u3_1qbit[2].real = cos(Phi)*sin_theta;
474  u3_1qbit[2].imag = sin(Phi)*sin_theta;
475  // the 2,2 element
476  u3_1qbit[3].real = cos(Phi+Lambda)*cos_theta;
477  u3_1qbit[3].imag = sin(Phi+Lambda)*cos_theta;
478 
479 
480  return 0;
481 
482 }
