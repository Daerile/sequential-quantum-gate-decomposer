452 int U3::calc_one_qubit_u3(double Theta, double Phi, double Lambda ) {
453 
454  double cos_theta = cos(Theta/2);
455  double sin_theta = sin(Theta/2);
456 
457  // the 1,1 element
458  u3_1qbit[0].real = cos_theta;
459  u3_1qbit[0].imag = 0;
460  // the 1,2 element
461  u3_1qbit[1].real = -cos(Lambda)*sin_theta;
462  u3_1qbit[1].imag = -sin(Lambda)*sin_theta;
463  // the 2,1 element
464  u3_1qbit[2].real = cos(Phi)*sin_theta;
465  u3_1qbit[2].imag = sin(Phi)*sin_theta;
466  // the 2,2 element
467  u3_1qbit[3].real = cos(Phi+Lambda)*cos_theta;
468  u3_1qbit[3].imag = sin(Phi+Lambda)*cos_theta;
469 
470 
471  return 0;
472 
473 }
