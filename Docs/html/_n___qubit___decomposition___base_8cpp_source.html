123 N_Qubit_Decomposition_Base::calc_decomposition_error(Matrix& decomposed_matrix ) {
124 
125  // (U-U_{approx}) (U-U_{approx})^\dagger = 2*I - U*U_{approx}^\dagger - U_{approx}*U^\dagger
126  // U*U_{approx}^\dagger = decomposed_matrix_copy
127 
128  Matrix A(matrix_size, matrix_size);
129  QGD_Complex16* A_data = A.get_data();
130  QGD_Complex16* decomposed_data = decomposed_matrix.get_data();
131  QGD_Complex16 phase;
132  phase.real = decomposed_matrix[0].real/(std::sqrt(decomposed_matrix[0].real*decomposed_matrix[0].real + decomposed_matrix[0].imag*decomposed_matrix[0].imag));
133  phase.imag = -decomposed_matrix[0].imag/(std::sqrt(decomposed_matrix[0].real*decomposed_matrix[0].real + decomposed_matrix[0].imag*decomposed_matrix[0].imag));
134 
135  for (int idx=0; idx<matrix_size; idx++ ) {
136  for (int jdx=0; jdx<matrix_size; jdx++ ) {
137 
138  if (idx==jdx) {
139  QGD_Complex16 mtx_val = mult(phase, decomposed_data[idx*matrix_size+jdx]);
140  A_data[idx*matrix_size+jdx].real = 2.0 - 2*mtx_val.real;
141  A_data[idx*matrix_size+jdx].imag = 0;
142  }
143  else {
144  QGD_Complex16 mtx_val_ij = mult(phase, decomposed_data[idx*matrix_size+jdx]);
145  QGD_Complex16 mtx_val_ji = mult(phase, decomposed_data[jdx*matrix_size+idx]);
146  A_data[idx*matrix_size+jdx].real = - mtx_val_ij.real - mtx_val_ji.real;
147  A_data[idx*matrix_size+jdx].imag = - mtx_val_ij.imag + mtx_val_ji.imag;
148  }
149 
150  }
151  }
152 
153 
154  Matrix alpha(matrix_size, 1);
155  Matrix beta(matrix_size, 1);
156  Matrix B = create_identity(matrix_size);
157 
158  // solve the generalized eigenvalue problem of I- 1/2
159  LAPACKE_zggev( CblasRowMajor, 'N', 'N',
160  matrix_size, A.get_data(), matrix_size, B.get_data(),
161  matrix_size, alpha.get_data(),
162  beta.get_data(), NULL, matrix_size, NULL,
163  matrix_size );
164 
165  // determine the largest eigenvalue
166  double eigval_max = 0;
167  for (int idx=0; idx<matrix_size; idx++) {
168  double eigval_abs = std::sqrt((alpha[idx].real*alpha[idx].real + alpha[idx].imag*alpha[idx].imag) / (beta[idx].real*beta[idx].real + beta[idx].imag*beta[idx].imag));
169  if ( eigval_max < eigval_abs ) eigval_max = eigval_abs;
170  }
171 
172  // the norm is the square root of the largest einegvalue.
173  decomposition_error = std::sqrt(eigval_max);
174 
175 
176 }

217 void N_Qubit_Decomposition_Base::solve_layer_optimization_problem( int num_of_parameters, gsl_vector *solution_guess_gsl) {
218 
219  if (gates.size() == 0 ) {
220  return;
221  }
222 
223 
224  if (solution_guess_gsl == NULL) {
225  solution_guess_gsl = gsl_vector_alloc(num_of_parameters);
226  }
227 
228 
229  if (optimized_parameters_mtx.size() == 0) {
230  optimized_parameters_mtx = Matrix_real(1, num_of_parameters);
231  memcpy(optimized_parameters_mtx.get_data(), solution_guess_gsl->data, num_of_parameters*sizeof(double) );
232  }
233 
234  // maximal number of iteration loops
235  int iteration_loops_max;
236  try {
237  iteration_loops_max = std::max(iteration_loops[qbit_num], 1);
238  }
239  catch (...) {
240  iteration_loops_max = 1;
241  }
242 
243 
244  // do the optimization loops
245  for (int idx=0; idx<iteration_loops_max; idx++) {
246 
247  size_t iter = 0;
248  int status;
249 
250  const gsl_multimin_fdfminimizer_type *T;
251  gsl_multimin_fdfminimizer *s;
252 
253  N_Qubit_Decomposition_Base* par = this;
254 
255 
256  gsl_multimin_function_fdf my_func;
257 
258 
259  my_func.n = num_of_parameters;
260  my_func.f = optimization_problem;
261  my_func.df = optimization_problem_grad;
262  my_func.fdf = optimization_problem_combined;
263  my_func.params = par;
264 
265 
266  T = gsl_multimin_fdfminimizer_vector_bfgs2;
267  s = gsl_multimin_fdfminimizer_alloc (T, num_of_parameters);
268 
269  gsl_multimin_fdfminimizer_set(s, &my_func, solution_guess_gsl, 0.01, 0.1);
270 
271  do {
272  iter++;
273  gsl_set_error_handler_off();
274  status = gsl_multimin_fdfminimizer_iterate (s);
275 
276  if (status) {
277  break;
278  }
279 
280  status = gsl_multimin_test_gradient (s->gradient, gradient_threshold);
281 
282  } while (status == GSL_CONTINUE && iter < iter_max);
283 
284  if (current_minimum > s->f) {
285  current_minimum = s->f;
286  memcpy( optimized_parameters_mtx.get_data(), s->x->data, num_of_parameters*sizeof(double) );
287  gsl_multimin_fdfminimizer_free (s);
288 
289  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
290  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
291  }
292  }
293  else {
294  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
295  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
296  }
297  gsl_multimin_fdfminimizer_free (s);
298  }
299 
300 
301 
302  }
