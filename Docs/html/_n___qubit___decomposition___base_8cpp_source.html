119 void N_Qubit_Decomposition_Base::solve_layer_optimization_problem( int num_of_parameters, gsl_vector *solution_guess_gsl) {
120 
121  if (operations.size() == 0 ) {
122  return;
123  }
124 
125 
126  if (solution_guess_gsl == NULL) {
127  solution_guess_gsl = gsl_vector_alloc(num_of_parameters);
128  }
129 
130  if (optimized_parameters == NULL) {
131  optimized_parameters = (double*)qgd_calloc(num_of_parameters,sizeof(double), 64);
132  memcpy(optimized_parameters, solution_guess_gsl->data, num_of_parameters*sizeof(double) );
133  }
134 
135  // maximal number of iteration loops
136  int iteration_loops_max;
137  try {
138  iteration_loops_max = std::max(iteration_loops[qbit_num], 1);
139  }
140  catch (...) {
141  iteration_loops_max = 1;
142  }
143 
144  // do the optimization loops
145  for (int idx=0; idx<iteration_loops_max; idx++) {
146 
147  size_t iter = 0;
148  int status;
149 
150  const gsl_multimin_fdfminimizer_type *T;
151  gsl_multimin_fdfminimizer *s;
152 
153  N_Qubit_Decomposition_Base* par = this;
154 
155 
156  gsl_multimin_function_fdf my_func;
157 
158 
159  my_func.n = num_of_parameters;
160  my_func.f = optimization_problem;
161  my_func.df = optimization_problem_grad;
162  my_func.fdf = optimization_problem_combined;
163  my_func.params = par;
164 
165 
166  T = gsl_multimin_fdfminimizer_vector_bfgs2;
167  s = gsl_multimin_fdfminimizer_alloc (T, num_of_parameters);
168 
169 
170  gsl_multimin_fdfminimizer_set (s, &my_func, solution_guess_gsl, 0.1, 0.1);
171 
172  do {
173  iter++;
174 
175  status = gsl_multimin_fdfminimizer_iterate (s);
176 
177  if (status) {
178  break;
179  }
180 
181  status = gsl_multimin_test_gradient (s->gradient, 1e-1);
182  /*if (status == GSL_SUCCESS) {
183  printf ("Minimum found\n");
184  }*/
185 
186  } while (status == GSL_CONTINUE && iter < 100);
187 
188  if (current_minimum > s->f) {
189  current_minimum = s->f;
190  memcpy( optimized_parameters, s->x->data, num_of_parameters*sizeof(double) );
191  gsl_multimin_fdfminimizer_free (s);
192 
193  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
194  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
195  }
196  }
197  else {
198  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
199  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
200  }
201  gsl_multimin_fdfminimizer_free (s);
202  }
203 
204 
205 
206  }
