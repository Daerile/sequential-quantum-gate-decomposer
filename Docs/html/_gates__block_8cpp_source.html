471 
472  // deriv_idx ... the index of the gate block for which the gradient is to be calculated
473  tbb::parallel_for( tbb::blocked_range<int>(0,gates.size()), [&](tbb::blocked_range<int> r) {
474  for (int deriv_idx=r.begin(); deriv_idx<r.end(); ++deriv_idx) {
475 
476 
477  Gate* gate_deriv = gates[deriv_idx];
478 
479  // for constant gate no gardient component is calculated
480  if ( gate_deriv->get_parameter_num() == 0 ) {
481  continue;
482  }
483 
484  int deriv_parameter_idx = 0;
485  for ( int idx=0; idx<deriv_idx; idx++ ) {
