128 UN::apply_to( Matrix_real& parameters, Matrix& input ) {
129 
130  if (input.rows != matrix_size ) {
131  std::stringstream sstream;
132  sstream << "Wrong matrix size in UN gate apply" << std::endl;
133  print(sstream, 0);
134  exit(-1);
135  }
136 
137  if (parameters.size() < parameter_num) {
138  std::stringstream sstream;
139  sstream << "Not enough parameters given for the UN gate" << std::endl;
140  print(sstream, 0);
141  exit(-1);
142  }
143 
144 
145  Matrix &&Umtx = get_submatrix( parameters );
146 
147  // get horizontal strided blocks of the input matrix
148  Matrix Block0 = Matrix( input.get_data(), input.rows/2, input.cols, input.stride );
149  Matrix Block1 = Matrix( input.get_data()+input.rows/2*input.stride, input.rows/2, input.cols, input.stride );
150 
151  // get the transformation of the blocks
152  Matrix Transformed_Block0 = dot( Umtx, Block0 );
153  Matrix Transformed_Block1 = dot( Umtx, Block1 );
154 
155  // put back the transformed data into input
156  memcpy( input.get_data(), Transformed_Block0.get_data(), Transformed_Block0.size()*sizeof(QGD_Complex16) );
157  memcpy( input.get_data()+input.rows/2*input.stride, Transformed_Block1.get_data(), Transformed_Block0.size()*sizeof(QGD_Complex16) );

228 void UN::reorder_qubits( std::vector<int> qbit_list ) {
229 
230  // check the number of qubits
231  if ((int)qbit_list.size() != qbit_num ) {
232  std::stringstream sstream;
233  sstream << "Wrong number of qubits" << std::endl;
234  print(sstream, 0);
235  }
236 
237 
238  int control_qbit_new = control_qbit;
239  int target_qbit_new = target_qbit;
240 
241  // setting the new value for the target qubit
242  for (int idx=0; idx<qbit_num; idx++) {
243  if (target_qbit == qbit_list[idx]) {
244  target_qbit_new = qbit_num-1-idx;
245  }
246  if (control_qbit == qbit_list[idx]) {
247  control_qbit_new = qbit_num-1-idx;
248  }
249  }
250 
251  control_qbit = control_qbit_new;
252  target_qbit = target_qbit_new;
253 }
