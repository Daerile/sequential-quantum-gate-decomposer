336  r"""
337  This method is called to create custom gate structure for the decomposition
338 
339  """
340 
341  from qgd_python.gates.qgd_Operation_Block import qgd_Operation_Block
342 
343  # creating an instance of the wrapper class qgd_Operation_Block
344  Operation_Block_ret = qgd_Operation_Block( qbit_num )
345 
346  control_qbit = qbit_num - 1
347 
348  for target_qbit in range(0, control_qbit ):
349 
350  # creating an instance of the wrapper class qgd_Operation_Block
351  Operation_Block_inner = qgd_Operation_Block( qbit_num )
352 
353  if target_qbit == 1:
354 
355  # add CNOT gate to the block
356  Operation_Block_inner.add_CNOT_To_End( 0, 1)
357 
358  # add U3 fate to the block
359  Theta = True
360  Phi = False
361  Lambda = True
362  Operation_Block_inner.add_U3_To_End( 0, Theta, Phi, Lambda )
363  Operation_Block_inner.add_U3_To_End( 1, Theta, Phi, Lambda )
364 
365  else:
366 
367  # add CNOT gate to the block
368  Operation_Block_inner.add_CNOT_To_End( control_qbit, target_qbit )
369 
370  # add U3 fate to the block
371  Theta = True
372  Phi = False
373  Lambda = True
374  Operation_Block_inner.add_U3_To_End( target_qbit, Theta, Phi, Lambda )
375  Operation_Block_inner.add_U3_To_End( control_qbit, Theta, Phi, Lambda )
376 
377 
378  Operation_Block_ret.add_Operation_Block_To_End( Operation_Block_inner )
