135 
136  for(std::vector<Operation*>::iterator it = operations.begin(); it != operations.end(); ++it) {
137 
138  Operation* operation = *it;
139  Matrix operation_mtx;
140 
141  if (operation->get_type() == CNOT_OPERATION) {
142  CNOT* cnot_operation = static_cast<CNOT*>(operation);
143  operation_mtx = cnot_operation->get_matrix();
144 
145  }
146  else if (operation->get_type() == U3_OPERATION) {
147  U3* u3_operation = static_cast<U3*>(operation);
148 
149  if (u3_operation->get_parameter_num() == 1 ) {
150  operation_mtx = u3_operation->get_matrix( parameters );
151  parameters = parameters + 1;
152  }
153  else if (u3_operation->get_parameter_num() == 2 ) {
154  #ifdef DEBUG

156  std::cout << "Operation_block::get_matrices: parameters contains NaN." << std::endl;
157  }
158 #endif
159  operation_mtx = u3_operation->get_matrix( parameters );
160  parameters = parameters + 2;
161  }
162  else if (u3_operation->get_parameter_num() == 3 ) {
163  operation_mtx = u3_operation->get_matrix( parameters );
164  parameters = parameters + 3;
165  }
166  else {
167  printf("The U3 operation has wrong number of parameters");
168  throw "The U3 operation has wrong number of parameters";
169  }
170 
171  }
172  else if (operation->get_type() == GENERAL_OPERATION) {
173  operation_mtx = operation->get_matrix();
174  }
175 
176 #ifdef DEBUG
177  if (operation_mtx.isnan()) {
178  std::cout << "Operation_block::get_matrices: operation_mtx contains NaN." << std::endl;

394 void Operation_block::list_operations( const double* parameters, int start_index ) {
395 
396  printf( "\nThe operations in the list of operations:\n" );
397 
398  int operation_idx = start_index;
399  int parameter_idx = parameter_num;
400 
401  for(int op_idx = operations.size()-1; op_idx>=0; op_idx--) {
402 
403  Operation* operation = operations[op_idx];
404 
405  if (operation->get_type() == CNOT_OPERATION) {
406  CNOT* cnot_operation = static_cast<CNOT*>(operation);
407 
408  printf( "%dth operation: CNOT with control qubit: %d and target qubit: %d\n", operation_idx, cnot_operation->get_control_qbit(), cnot_operation->get_target_qbit() );
409  operation_idx = operation_idx + 1;
410  }
411  else if (operation->get_type() == U3_OPERATION) {
412 
413  // definig the U3 parameters
414  double vartheta;
415  double varphi;
416  double varlambda;
417 
418  // get the inverse parameters of the U3 rotation
419 
420  U3* u3_operation = static_cast<U3*>(operation);
421 
422  if ((u3_operation->get_parameter_num() == 1) && u3_operation->is_theta_parameter()) {
423  vartheta = std::fmod( parameters[parameter_idx-1], 4*M_PI);
424  varphi = 0;
425  varlambda =0;
426  parameter_idx = parameter_idx - 1;
427 
428  }
429  else if ((u3_operation->get_parameter_num() == 1) && u3_operation->is_phi_parameter()) {
430  vartheta = 0;
431  varphi = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
432  varlambda =0;
433  parameter_idx = parameter_idx - 1;
434  }
435  else if ((u3_operation->get_parameter_num() == 1) && u3_operation->is_lambda_parameter()) {
436  vartheta = 0;
437  varphi = 0;
438  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
439  parameter_idx = parameter_idx - 1;
440  }
441  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_theta_parameter() && u3_operation->is_phi_parameter() ) {
442  vartheta = std::fmod( parameters[ parameter_idx-2 ], 4*M_PI);
443  varphi = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
444  varlambda = 0;
445  parameter_idx = parameter_idx - 2;
446  }
447  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_theta_parameter() && u3_operation->is_lambda_parameter() ) {
448  vartheta = std::fmod( parameters[ parameter_idx-2 ], 4*M_PI);
449  varphi = 0;
450  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
451  parameter_idx = parameter_idx - 2;
452  }
453  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_phi_parameter() && u3_operation->is_lambda_parameter() ) {
454  vartheta = 0;
455  varphi = std::fmod( parameters[ parameter_idx-2], 2*M_PI);
456  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
457  parameter_idx = parameter_idx - 2;
458  }
459  else if ((u3_operation->get_parameter_num() == 3)) {
460  vartheta = std::fmod( parameters[ parameter_idx-3 ], 4*M_PI);
461  varphi = std::fmod( parameters[ parameter_idx-2 ], 2*M_PI);
462  varlambda = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
463  parameter_idx = parameter_idx - 3;
464  }
465 
466 // message = message + "U3 on target qubit %d with parameters theta = %f, phi = %f and lambda = %f";
467  printf("%dth operation: U3 on target qubit: %d and with parameters theta = %f, phi = %f and lambda = %f\n", operation_idx, u3_operation->get_target_qbit(), vartheta, varphi, varlambda );
468  operation_idx = operation_idx + 1;
469 
470  }
471  else if (operation->get_type() == BLOCK_OPERATION) {
472  Operation_block* block_operation = static_cast<Operation_block*>(operation);
