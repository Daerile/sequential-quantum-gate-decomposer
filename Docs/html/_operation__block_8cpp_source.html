507  }

516

518  operation_idx = operation_idx + 1;
519  }
520  else if (operation->get_type() == U3_OPERATION) {
521 
522  // definig the U3 parameters

545  vartheta = 0;
546  varphi = 0;

548  parameter_idx = parameter_idx - 1;
549  }
550  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_theta_parameter() && u3_operation->is_phi_parameter() ) {
551  vartheta = std::fmod( parameters[ parameter_idx-2 ], 4*M_PI);
552  varphi = std::fmod( parameters[ parameter_idx-1 ], 2*M_PI);
553  varlambda = 0;
554  parameter_idx = parameter_idx - 2;
555  }
556  else if ((u3_operation->get_parameter_num() == 2) && u3_operation->is_theta_parameter() && u3_operation->is_lambda_parameter() ) {
557  vartheta = std::fmod( parameters[ parameter_idx-2 ], 4*M_PI);
