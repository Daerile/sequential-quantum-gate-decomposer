38 void *aligned_alloc(size_t alignment, size_t size, bool zero) {
39 
40  size_t request_size = size + alignment;
41  char* buf = (char*)malloc(request_size);
42 
43  if (zero) {
44  memset( buf, 0, request_size );
45  }
46 
47  size_t remainder = ((size_t)buf) % alignment;
48  size_t offset = alignment - remainder;
49  char* ret = buf + (unsigned char)offset;
50 
51  // store how many extra bytes we allocated in the byte just before the
52  // pointer we return
53  *(unsigned char*)(ret - 1) = offset;
54 
55  return (void*)ret;
56 }
