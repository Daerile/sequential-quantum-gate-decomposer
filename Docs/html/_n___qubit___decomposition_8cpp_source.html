232 
233 
234  if (decomposition_finalized) {
235  std::stringstream sstream;

237  print(sstream, 1);

243  }

245  // obtaining the subdecomposed submatrices
246 //Matrix_real optimized_parameters_mtx_tmp(optimized_parameters, 1, parameter_num );
247  Matrix subdecomposed_matrix_mtx = get_transformed_matrix( optimized_parameters_mtx, gates.begin(), gates.size(), Umtx );
248  QGD_Complex16* subdecomposed_matrix = subdecomposed_matrix_mtx.get_data();
249 
250  // get the most unitary submatrix
251  // get the number of 2qubit submatrices
252  int submatrices_num_row = 2;
253 
254  // get the size of the submatrix
255  int submatrix_size = int(matrix_size/2);
256 
257  // fill up the submatrices and select the most unitary submatrix
258 
259  Matrix most_unitary_submatrix_mtx = Matrix(submatrix_size, submatrix_size );
260  QGD_Complex16* most_unitary_submatrix = most_unitary_submatrix_mtx.get_data();
261  double unitary_error_min = 1e8;
262 
263  for (int idx=0; idx<submatrices_num_row; idx++) { // in range(0,submatrices_num_row):
264  for (int jdx=0; jdx<submatrices_num_row; jdx++) { // in range(0,submatrices_num_row):
265 
266  Matrix submatrix_mtx = Matrix(submatrix_size, submatrix_size);
267  QGD_Complex16* submatrix = submatrix_mtx.get_data();
268 
269  for ( int row_idx=0; row_idx<submatrix_size; row_idx++ ) {
270  int matrix_offset = idx*(matrix_size*submatrix_size) + jdx*(submatrix_size) + row_idx*matrix_size;
271  int submatrix_offset = row_idx*submatrix_size;
272  memcpy(submatrix+submatrix_offset, subdecomposed_matrix+matrix_offset, submatrix_size*sizeof(QGD_Complex16));
273  }
274 
275  // calculate the product of submatrix*submatrix'
276  Matrix submatrix_mtx_adj = submatrix_mtx;
277  submatrix_mtx_adj.transpose();
278  submatrix_mtx_adj.conjugate();
279  Matrix submatrix_prod = dot( submatrix_mtx, submatrix_mtx_adj);
280 
281  // subtract corner element
282  QGD_Complex16 corner_element = submatrix_prod[0];
283  for (int row_idx=0; row_idx<submatrix_size; row_idx++) {
284  submatrix_prod[row_idx*submatrix_size+row_idx].real = submatrix_prod[row_idx*submatrix_size+row_idx].real - corner_element.real;
285  submatrix_prod[row_idx*submatrix_size+row_idx].imag = submatrix_prod[row_idx*submatrix_size+row_idx].imag - corner_element.imag;
286  }
287 
288  double unitary_error = cblas_dznrm2( submatrix_size*submatrix_size, submatrix_prod.get_data(), 1 );
289 
290  if (unitary_error < unitary_error_min) {
291  unitary_error_min = unitary_error;
292  memcpy(most_unitary_submatrix, submatrix, submatrix_size*submatrix_size*sizeof(QGD_Complex16));
293  }
294 
295  }
296  }
297 
298  // if the qubit number in the submatirx is greater than 2 new N-qubit decomposition is started
299 
300  // create class tp decompose submatrices
301  N_Qubit_Decomposition* cdecomposition = new N_Qubit_Decomposition(most_unitary_submatrix_mtx, qbit_num-1, optimize_layer_num, initial_guess);
302 
303  // setting the verbosity
304  cdecomposition->set_verbose( verbose );
305 
306  // Maximal number of iteartions in the optimization process
307  cdecomposition->set_max_iteration(max_iterations);
308 
309  // Set the number of identical successive blocks in the sub-decomposition
310  cdecomposition->set_identical_blocks(identical_blocks);
311 
312  // Set the maximal number of layers for the sub-decompositions
313  cdecomposition->set_max_layer_num(max_layer_num);
314 
315  // setting the iteration loops in each step of the optimization process
316  cdecomposition->set_iteration_loops( iteration_loops );
317 
318  // setting gate layer
319  cdecomposition->set_optimization_blocks( optimization_block );
320 
321  // set custom gate structure if given
322  cdecomposition->set_custom_gate_structure( gate_structure );
323 
324  // set the toleration of the optimization process
325  cdecomposition->set_optimization_tolerance( optimization_tolerance );
326 
327  // starting the decomposition of the random unitary
328  cdecomposition->start_decomposition(false, false);
329 
330 
331  // saving the decomposition gates
332  extract_subdecomposition_results( reinterpret_cast<Sub_Matrix_Decomposition*>(cdecomposition) );
333 
334  delete cdecomposition;

475 void
476 N_Qubit_Decomposition::simplify_layers() {
477 
478  //The stringstream input to store the output messages.
479  std::stringstream sstream;
480  sstream << "***************************************************************" << std::endl;
481  sstream << "Try to simplify layers" << std::endl;
482  sstream << "***************************************************************" << std::endl;
483  print(sstream, 1);
484 
485 
486 
487  // current starting index of the optimized parameters
488  int parameter_idx = 0;
489 
490  Gates_block* gates_loc = new Gates_block( qbit_num );
491  Matrix_real optimized_parameters_loc_mtx(1, parameter_num);
492  double* optimized_parameters_loc = optimized_parameters_loc_mtx.get_data();
493  unsigned int parameter_num_loc = 0;
494 
495  unsigned int layer_idx = 0;
496 
497  while (layer_idx < gates.size()) {
498 
499  // generate a block of gates to be simplified
500  // (containg only successive two-qubit gates)
501  Gates_block* block_to_simplify = new Gates_block( qbit_num );
502 
503  std::vector<int> involved_qbits;
504  // layers in the block to be simplified
505  Gates_block* blocks_to_save = new Gates_block( qbit_num );
506 
507  // get the successive gates involving the same qubits
508  while (true) {
509 
510  if (layer_idx >= gates.size() ) {
511  break;
512  }
513 
514  // get the current layer of gates
515  Gate* gate = gates[layer_idx];
516  layer_idx = layer_idx + 1;
517 
518  Gates_block* block_gate;
519  if (gate->get_type() == BLOCK_OPERATION) {
520  block_gate = static_cast<Gates_block*>(gate);
521  }
522  else {
523  layer_idx = layer_idx + 1;
524  continue;
525  }
526 
527 
528  //get the involved qubits
529  std::vector<int> involved_qbits_op = block_gate->get_involved_qubits();
530 
531  // add involved qubits of the gate to the list of involved qubits in the layer
532  for (std::vector<int>::iterator it=involved_qbits_op.begin(); it!=involved_qbits_op.end(); it++) {
533  add_unique_elelement( involved_qbits, *it );
534  }
535 
536  if ( (involved_qbits.size())> 2 && blocks_to_save->get_gate_num() > 0 ) {
537  layer_idx = layer_idx -1;
538  break;
539  }
540 
541  blocks_to_save->combine(block_gate);
542 
543  // adding the gates to teh block if they act on the same qubits
544  block_to_simplify->combine(block_gate);
545 
546 
547  }
548 
549  //number of perations in the block
550  unsigned int parameter_num_block = block_to_simplify->get_parameter_num();
551 
552  // get the number of two-qubit gates and store the block gates if the number of CNOT gates cannot be reduced
553  gates_num gate_nums = block_to_simplify->get_gate_nums();
554 
555  if (gate_nums.cnot + gate_nums.cz + gate_nums.ch < 2 || involved_qbits.size()> 2) {
556  gates_loc->combine(blocks_to_save);
557  memcpy(optimized_parameters_loc+parameter_num_loc, optimized_parameters_mtx.get_data()+parameter_idx, parameter_num_block*sizeof(double) );
558  parameter_idx = parameter_idx + parameter_num_block;
559  parameter_num_loc = parameter_num_loc + parameter_num_block;
560 
561  if ( block_to_simplify != NULL ) {
562  delete block_to_simplify;
563  block_to_simplify = NULL;
564  }
565 
566  if ( blocks_to_save != NULL ) {
567  delete blocks_to_save;
568  blocks_to_save = NULL;
569  }
570 
571  involved_qbits.clear();
572 
573  continue;
574  }
575 
576 
577  involved_qbits.clear();
578 
579  // simplify the given layer
580  std::map<int,int> max_layer_num_loc;
581  max_layer_num_loc.insert( std::pair<int, int>(2, gate_nums.cnot+gate_nums.cz+gate_nums.ch-1 ) );
582  Gates_block* simplified_layer = NULL;
583  double* simplified_parameters = NULL;
584  unsigned int simplified_parameter_num=0;
585 
586  // Try to simplify the sequence of 2-qubit gates
587  int simplification_status = simplify_layer( block_to_simplify, optimized_parameters_mtx.get_data()+parameter_idx, parameter_num_block, max_layer_num_loc, simplified_layer, simplified_parameters, simplified_parameter_num );
588 
589 
590 
591  // adding the simplified gates (or the non-simplified if the simplification was not successfull)
592  if (simplification_status == 0) {
593  gates_loc->combine( simplified_layer );
594 
595  if (parameter_num < parameter_num_loc + simplified_parameter_num ) {
596  //optimized_parameters_loc = (double*)qgd_realloc( optimized_parameters_loc, parameter_num_loc + simplified_parameter_num, sizeof(double), 64 );
597  optimized_parameters_loc_mtx = Matrix_real(1, parameter_num_loc + simplified_parameter_num);
598  optimized_parameters_loc = optimized_parameters_loc_mtx.get_data();
599  }
600  memcpy(optimized_parameters_loc+parameter_num_loc, simplified_parameters, simplified_parameter_num*sizeof(double) );
601  parameter_num_loc = parameter_num_loc + simplified_parameter_num;
602  }
603  else {
604  // addign the stacked gate to the list, sice the simplification was unsuccessful
605  gates_loc->combine( blocks_to_save );
606 
607  if (parameter_num < parameter_num_loc + parameter_num_block ) {
608  //optimized_parameters_loc = (double*)qgd_realloc( optimized_parameters_loc, parameter_num_loc + parameter_num_block, sizeof(double), 64 );
609  optimized_parameters_loc_mtx = Matrix_real(1, parameter_num_loc + parameter_num_block);

688 N_Qubit_Decomposition::simplify_layer( Gates_block* layer, double* parameters, unsigned int parameter_num_block, std::map<int,int> max_layer_num_loc, Gates_block* &simplified_layer, double* &simplified_parameters, unsigned int &simplified_parameter_num) {
689 
690  //The stringstream input to store the output messages.
691  std::stringstream sstream;
692  sstream << "Try to simplify sub-structure " << std::endl;
693  print(sstream, 1);
694 
695 
696 
697  // get the target bit
698  int target_qbit = -1;
699  int control_qbit = -1;
700 
701  std::vector<Gate*> layer_gates = layer->get_gates();
702  for (std::vector<Gate*>::iterator it = layer_gates.begin(); it!=layer_gates.end(); it++) {
703  Gate* op = *it;
704  if (op->get_type() == CNOT_OPERATION || op->get_type() == CZ_OPERATION || op->get_type() == CH_OPERATION) {
705  target_qbit = op->get_target_qbit();
706  control_qbit = op->get_control_qbit();
707  break;
708  }
709  }
710 
711  // if there are no target or control qubits, return the initial values
712  if ( (target_qbit == -1) || ( control_qbit == -1 ) ) {
713  return 1;
714  }
715 
716  // get the matrix of the two qubit space
717 
718  // reorder the control and target qubits to the end of the list
719  std::vector<int> qbits_reordered;
720  for (int qbit_idx=qbit_num-1; qbit_idx>-1; qbit_idx-- ) { // in range(self.qbit_num-1,-1,-1):
721  if ( (qbit_idx != target_qbit) && (qbit_idx != control_qbit) ) {
722  qbits_reordered.push_back(qbit_idx);
723  }
724  }
725 
726  qbits_reordered.push_back(control_qbit);
727  qbits_reordered.push_back(target_qbit);
728 
729 
730  // construct abstarct gates correspond to the reordeerd qubits
731  Gates_block* reordered_layer = layer->clone();
732  reordered_layer->reorder_qubits( qbits_reordered );
733 
734  // get the reordered N-qubit matrix of the layer
735  Matrix_real parameters_mtx( parameters, 1, parameter_num );
736  Matrix full_matrix_reordered = reordered_layer->get_matrix( parameters_mtx );
737  delete reordered_layer;
738 
739 
740 
741 
742 
743 
744  // construct the Two-qubit submatrix from the reordered matrix
745  Matrix submatrix = Matrix(4,4);
746  for ( int element_idx=0; element_idx<16; element_idx++) {
747  int col_idx = element_idx % 4;
748  int row_idx = int((element_idx-col_idx)/4);
749  submatrix[element_idx].real = full_matrix_reordered[col_idx*matrix_size+row_idx].real;
750  submatrix[element_idx].imag = -full_matrix_reordered[col_idx*matrix_size+row_idx].imag;
751  }
752 
753  // decompose the chosen 2-qubit unitary
754  N_Qubit_Decomposition* cdecomposition = new N_Qubit_Decomposition(submatrix, 2, true, initial_guess);
755 
756  // set the maximal number of layers
757  cdecomposition->set_max_layer_num( max_layer_num_loc );
758 
759  // suppress output messages
760  cdecomposition->set_verbose( false );
761 
762  // starting the decomposition
763  cdecomposition->start_decomposition(true, false);
764 
765 
766 
767  // check whether simplification was succesfull
768  if (!cdecomposition->check_optimization_solution()) {
769  // return with the original layer, if the simplification wa snot successfull
770  std::stringstream sstream;
771  sstream << "The simplification of the sub-structure was not possible" << std::endl;
772  print(sstream, 1);
773  delete cdecomposition;
774  return 1;
775  }
776 
777 
778 
779  // contruct the layer containing the simplified gates in the N-qubit space
780  // but first get the inverse reordered qubit list
781  std::vector<int> qbits_inverse_reordered;
782  for (int idx=0; idx<qbit_num; idx++) {
783  qbits_inverse_reordered.push_back(-1);
784  }
785 
786  for (int qbit_idx=qbit_num-1; qbit_idx>-1; qbit_idx-- ) { // in range(self.qbit_num-1,-1,-1):
787  qbits_inverse_reordered[qbit_num-1-qbits_reordered[qbit_num-1-qbit_idx]] = qbit_idx;
788  }
789 
790  simplified_layer = new Gates_block(qbit_num);
791 
792  std::vector<Gate*> simplifying_gates = cdecomposition->get_gates();
793  for ( std::vector<Gate*>::iterator it=simplifying_gates.begin(); it!=simplifying_gates.end(); it++) { //int gate_block_idx=0 in range(0,len(cdecomposition.gates)):
794  Gate* op = *it;
795 
796  Gates_block* block_op = static_cast<Gates_block*>( op );
797  block_op->set_qbit_num( qbit_num );
798  block_op->reorder_qubits( qbits_inverse_reordered );
799  simplified_layer->combine( block_op );
800  }
801 
802 
803 
804  simplified_parameter_num = cdecomposition->get_parameter_num();
805  simplified_parameters = (double*)qgd_calloc(simplified_parameter_num, sizeof(double), 64);
806  cdecomposition->get_optimized_parameters( simplified_parameters );
807 
808 
809  gates_num gate_nums_layer = layer->get_gate_nums();
810  gates_num gate_nums_simplified = simplified_layer->get_gate_nums();
811 
812  sstream.str("");
813  sstream << gate_nums_layer.cnot+gate_nums_layer.cz+gate_nums_layer.ch << " two-qubit gates successfully simplified to " << gate_nums_simplified.cnot << " CNOT gates" << std::endl;
814  print(sstream,1);
815 
816 
817  //release allocated memory
818  delete cdecomposition;
819 
820 
821  return 0;
