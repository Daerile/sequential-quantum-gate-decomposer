250 void N_Qubit_Decomposition::decompose_submatrix() {
251 
252  if (decomposition_finalized) {
253  if (verbose) {
254  printf("Decomposition was already finalized\n");
255  }
256  return;
257  }
258 
259  if (qbit_num == 2) {
260  return;
261  }
262 
263  // obtaining the subdecomposed submatrices
264  Matrix subdecomposed_matrix_mtx = get_transformed_matrix( optimized_parameters, operations.begin(), operations.size(), Umtx );
265  QGD_Complex16* subdecomposed_matrix = subdecomposed_matrix_mtx.get_data();
266 
267  // get the most unitary submatrix
268  // get the number of 2qubit submatrices
269  int submatrices_num_row = 2;
270 
271  // get the size of the submatrix
272  int submatrix_size = int(matrix_size/2);
273 
274  // fill up the submatrices and select the most unitary submatrix
275 
276  Matrix most_unitary_submatrix_mtx = Matrix(submatrix_size, submatrix_size );
277  QGD_Complex16* most_unitary_submatrix = most_unitary_submatrix_mtx.get_data();
278  double unitary_error_min = 1e8;
279 
280  for (int idx=0; idx<submatrices_num_row; idx++) { // in range(0,submatrices_num_row):
281  for (int jdx=0; jdx<submatrices_num_row; jdx++) { // in range(0,submatrices_num_row):
282 
283  Matrix submatrix_mtx = Matrix(submatrix_size, submatrix_size);
284  QGD_Complex16* submatrix = submatrix_mtx.get_data();
285 
286  for ( int row_idx=0; row_idx<submatrix_size; row_idx++ ) {
287  int matrix_offset = idx*(matrix_size*submatrix_size) + jdx*(submatrix_size) + row_idx*matrix_size;
288  int submatrix_offset = row_idx*submatrix_size;
289  memcpy(submatrix+submatrix_offset, subdecomposed_matrix+matrix_offset, submatrix_size*sizeof(QGD_Complex16));
290  }
291 
292  // check whether the matrix is not null-matrix
293  double submatrix_norm = cblas_dznrm2( submatrix_size*submatrix_size, submatrix, 1 );
294  if (submatrix_norm < 1e-6) continue;
295 
296  // calculate the product of submatrix*submatrix'
297  Matrix submatrix_mtx_adj = submatrix_mtx;
298  submatrix_mtx_adj.transpose();
299  submatrix_mtx_adj.conjugate();
300  Matrix submatrix_prod = dot( submatrix_mtx, submatrix_mtx_adj);
301 
302  // subtract corner element
303  QGD_Complex16 corner_element = submatrix_prod[0];
304  for (int row_idx=0; row_idx<submatrix_size; row_idx++) {
305  submatrix_prod[row_idx*submatrix_size+row_idx].real = submatrix_prod[row_idx*submatrix_size+row_idx].real - corner_element.real;
306  submatrix_prod[row_idx*submatrix_size+row_idx].imag = submatrix_prod[row_idx*submatrix_size+row_idx].imag - corner_element.imag;
307  }
308 
309  double unitary_error = cblas_dznrm2( submatrix_size*submatrix_size, submatrix_prod.get_data(), 1 );
310 
311  if (unitary_error < unitary_error_min) {
312  unitary_error_min = unitary_error;
313  memcpy(most_unitary_submatrix, submatrix, submatrix_size*submatrix_size*sizeof(QGD_Complex16));
314  }
315 
316  }
317  }
318 
319  // if the qubit number in the submatirx is greater than 2 new N-qubit decomposition is started
320 
321  // create class tp decompose submatrices
322  N_Qubit_Decomposition* cdecomposition = new N_Qubit_Decomposition(most_unitary_submatrix_mtx, qbit_num-1, optimize_layer_num, initial_guess);
323 
324  // setting the verbosity
325  cdecomposition->set_verbose( verbose );
326 
327 
328  // Maximal number of iteartions in the optimization process
329  cdecomposition->set_max_iteration(max_iterations);
330 
331  // Set the number of identical successive blocks in the sub-decomposition
332  cdecomposition->set_identical_blocks(identical_blocks);
333 
334  // Set the maximal number of layers for the sub-decompositions
335  cdecomposition->set_max_layer_num(max_layer_num);
336 
337  // setting the iteration loops in each step of the optimization process
338  cdecomposition->set_iteration_loops( iteration_loops );
339 
340  // setting operation layer
341  cdecomposition->set_optimization_blocks( optimization_block );
342 
343  // starting the decomposition of the random unitary
344  cdecomposition->start_decomposition(false, false);
345 
346 
347  // saving the decomposition operations
348  extract_subdecomposition_results( reinterpret_cast<Sub_Matrix_Decomposition*>(cdecomposition) );
349 
350  delete cdecomposition;

454

493 
494  } while (status == GSL_CONTINUE && iter < 100);

566 void N_Qubit_Decomposition::simplify_layers() {
567 
568  if (verbose) {
569  printf("***************************************************************\n");
570  printf("Try to simplify layers\n");
571  printf("***************************************************************\n");
572  }
573 
574  // current starting index of the optimized parameters
575  int parameter_idx = 0;
576 
577  Operation_block* operations_loc = new Operation_block( qbit_num );
578  double* optimized_parameters_loc = (double*)qgd_calloc(parameter_num, sizeof(double), 64);
579  unsigned int parameter_num_loc = 0;
580 
581  unsigned int layer_idx = 0;
582 
583  while (layer_idx < operations.size()) {
584 
585  // generate a block of operations to be simplified
586  // (containg only successive two-qubit operations)
587  Operation_block* block_to_simplify = new Operation_block( qbit_num );
588 
589  std::vector<int> involved_qbits;
590  // layers in the block to be simplified
591  Operation_block* blocks_to_save = new Operation_block( qbit_num );
592 
593  // get the successive operations involving the same qubits
594  while (true) {
595 
596  if (layer_idx >= operations.size() ) {
597  break;
598  }
599 
600  // get the current layer of operations
601  Operation* operation = operations[layer_idx];
602  layer_idx = layer_idx + 1;
603 
604  Operation_block* block_operation;
605  if (operation->get_type() == BLOCK_OPERATION) {
606  block_operation = static_cast<Operation_block*>(operation);
607  }
608  else {
609  layer_idx = layer_idx + 1;
610  continue;
611  }
612 
613 
614  //get the involved qubits
615  std::vector<int> involved_qbits_op = block_operation->get_involved_qubits();
616 
617  // add involved qubits of the operation to the list of involved qubits in the layer
618  for (std::vector<int>::iterator it=involved_qbits_op.begin(); it!=involved_qbits_op.end(); it++) {
619  add_unique_elelement( involved_qbits, *it );
620  }
621 
622  if ( (involved_qbits.size())> 2 && blocks_to_save->get_operation_num() > 0 ) {
623  layer_idx = layer_idx -1;
624  involved_qbits.clear();
625  break;
626  }
627 
628  blocks_to_save->combine(block_operation);
629 
630  // adding the operations to teh block if they act on the same qubits
631  block_to_simplify->combine(block_operation);
632 
633 
634  }
635 
636  //number of perations in the block
637  unsigned int parameter_num_block = block_to_simplify->get_parameter_num();
638 
639  // get the number of CNOT gates and store the block operations if the number of CNOT gates cannot be reduced
640  gates_num gate_nums = block_to_simplify->get_gate_nums();
641 
642  if (gate_nums.cnot < 2) {
643  operations_loc->combine(blocks_to_save);
644  memcpy(optimized_parameters_loc+parameter_num_loc, optimized_parameters+parameter_idx, parameter_num_block*sizeof(double) );
645  parameter_idx = parameter_idx + parameter_num_block;
646  parameter_num_loc = parameter_num_loc + parameter_num_block;
647 
648  if ( block_to_simplify != NULL ) {
649  delete block_to_simplify;
650  block_to_simplify = NULL;
651  }
652 
653  if ( blocks_to_save != NULL ) {
654  delete blocks_to_save;
655  blocks_to_save = NULL;
656  }
657 
658  continue;
659  }
660 
661 
662  // simplify the given layer
663  std::map<int,int> max_layer_num_loc;
664  max_layer_num_loc.insert( std::pair<int, int>(2, gate_nums.cnot-1 ) );
665  Operation_block* simplified_layer = NULL;
666  double* simplified_parameters = NULL;
667  unsigned int simplified_parameter_num=0;
668 
669  // Try to simplify the sequence of 2-qubit operations
670  int simplification_status = simplify_layer( block_to_simplify, optimized_parameters+parameter_idx, parameter_num_block, max_layer_num_loc, simplified_layer, simplified_parameters, simplified_parameter_num );
671 
672 
673 
674  // adding the simplified operations (or the non-simplified if the simplification was not successfull)
675  if (simplification_status == 0) {
676  operations_loc->combine( simplified_layer );
677 
678  if (parameter_num < parameter_num_loc + simplified_parameter_num ) {
679  optimized_parameters_loc = (double*)qgd_realloc( optimized_parameters_loc, parameter_num_loc + simplified_parameter_num, sizeof(double), 64 );
680  }
681  memcpy(optimized_parameters_loc+parameter_num_loc, simplified_parameters, simplified_parameter_num*sizeof(double) );
682  parameter_num_loc = parameter_num_loc + simplified_parameter_num;
683  }
684  else {
685  // addign the stacked operation to the list, sice the simplification was unsuccessful
686  operations_loc->combine( blocks_to_save );
687 
688  if (parameter_num < parameter_num_loc + parameter_num_block ) {
689  optimized_parameters_loc = (double*)qgd_realloc( optimized_parameters_loc, parameter_num_loc + parameter_num_block, sizeof(double), 64 );

786  return 1;
787  }

790

792  std::vector<int> qbits_reordered;
793  for (int qbit_idx=qbit_num-1; qbit_idx>-1; qbit_idx-- ) { // in range(self.qbit_num-1,-1,-1):

795  qbits_reordered.push_back(qbit_idx);

819  int row_idx = int((element_idx-col_idx)/4);

821  submatrix[element_idx].imag = -full_matrix_reordered[col_idx*matrix_size+row_idx].imag;
822  }
