230 N_Qubit_Decomposition::calc_decomposition_error(Matrix& decomposed_matrix ) {
231 
232  // (U-U_{approx}) (U-U_{approx})^\dagger = 2*I - U*U_{approx}^\dagger - U_{approx}*U^\dagger
233  // U*U_{approx}^\dagger = decomposed_matrix_copy
234 
235  Matrix A(matrix_size, matrix_size);
236  QGD_Complex16* A_data = A.get_data();
237  QGD_Complex16* decomposed_data = decomposed_matrix.get_data();
238  QGD_Complex16 phase;
239  phase.real = decomposed_matrix[0].real/(std::sqrt(decomposed_matrix[0].real*decomposed_matrix[0].real + decomposed_matrix[0].imag*decomposed_matrix[0].imag));
240  phase.imag = -decomposed_matrix[0].imag/(std::sqrt(decomposed_matrix[0].real*decomposed_matrix[0].real + decomposed_matrix[0].imag*decomposed_matrix[0].imag));
241 
242  for (int idx=0; idx<matrix_size; idx++ ) {
243  for (int jdx=0; jdx<matrix_size; jdx++ ) {
244 
245  if (idx==jdx) {
246  QGD_Complex16 mtx_val = mult(phase, decomposed_data[idx*matrix_size+jdx]);
247  A_data[idx*matrix_size+jdx].real = 2.0 - 2*mtx_val.real;
248  A_data[idx*matrix_size+jdx].imag = 0;
249  }
250  else {
251  QGD_Complex16 mtx_val_ij = mult(phase, decomposed_data[idx*matrix_size+jdx]);
252  QGD_Complex16 mtx_val_ji = mult(phase, decomposed_data[jdx*matrix_size+idx]);
253  A_data[idx*matrix_size+jdx].real = - mtx_val_ij.real - mtx_val_ji.real;
254  A_data[idx*matrix_size+jdx].imag = - mtx_val_ij.imag + mtx_val_ji.imag;
255  }
256 
257  }
258  }
259 
260 
261  Matrix alpha(matrix_size, 1);
262  Matrix beta(matrix_size, 1);
263  Matrix B = create_identity(matrix_size);
264 
265  // solve the generalized eigenvalue problem of I- 1/2
266  LAPACKE_zggev( CblasRowMajor, 'N', 'N',
267  matrix_size, A.get_data(), matrix_size, B.get_data(),
268  matrix_size, alpha.get_data(),
269  beta.get_data(), NULL, matrix_size, NULL,
270  matrix_size );
271 
272  // determine the largest eigenvalue
273  double eigval_max = 0;
274  for (int idx=0; idx<matrix_size; idx++) {
275  double eigval_abs = std::sqrt((alpha[idx].real*alpha[idx].real + alpha[idx].imag*alpha[idx].imag) / (beta[idx].real*beta[idx].real + beta[idx].imag*beta[idx].imag));
276  if ( eigval_max < eigval_abs ) eigval_max = eigval_abs;
277  }
278 
279  // the norm is the square root of the largest einegvalue.
280  decomposition_error = std::sqrt(eigval_max);
281

291 N_Qubit_Decomposition::decompose_submatrix() {
292 
293  if (decomposition_finalized) {
294  if (verbose) {
295  printf("Decomposition was already finalized\n");
296  }
297  return;
298  }
299 
300  if (qbit_num == 2) {
301  return;
302  }
303 
304  // obtaining the subdecomposed submatrices
305  Matrix subdecomposed_matrix_mtx = get_transformed_matrix( optimized_parameters, gates.begin(), gates.size(), Umtx );
306  QGD_Complex16* subdecomposed_matrix = subdecomposed_matrix_mtx.get_data();
307 
308  // get the most unitary submatrix
309  // get the number of 2qubit submatrices
310  int submatrices_num_row = 2;
311 
312  // get the size of the submatrix
313  int submatrix_size = int(matrix_size/2);
314 
315  // fill up the submatrices and select the most unitary submatrix
316 
317  Matrix most_unitary_submatrix_mtx = Matrix(submatrix_size, submatrix_size );
318  QGD_Complex16* most_unitary_submatrix = most_unitary_submatrix_mtx.get_data();
319  double unitary_error_min = 1e8;
320 
321  for (int idx=0; idx<submatrices_num_row; idx++) { // in range(0,submatrices_num_row):
322  for (int jdx=0; jdx<submatrices_num_row; jdx++) { // in range(0,submatrices_num_row):
323 
324  Matrix submatrix_mtx = Matrix(submatrix_size, submatrix_size);
325  QGD_Complex16* submatrix = submatrix_mtx.get_data();
326 
327  for ( int row_idx=0; row_idx<submatrix_size; row_idx++ ) {
328  int matrix_offset = idx*(matrix_size*submatrix_size) + jdx*(submatrix_size) + row_idx*matrix_size;
329  int submatrix_offset = row_idx*submatrix_size;
330  memcpy(submatrix+submatrix_offset, subdecomposed_matrix+matrix_offset, submatrix_size*sizeof(QGD_Complex16));
331  }
332 
333  // calculate the product of submatrix*submatrix'
334  Matrix submatrix_mtx_adj = submatrix_mtx;
335  submatrix_mtx_adj.transpose();
336  submatrix_mtx_adj.conjugate();
337  Matrix submatrix_prod = dot( submatrix_mtx, submatrix_mtx_adj);
338 
339  // subtract corner element
340  QGD_Complex16 corner_element = submatrix_prod[0];
341  for (int row_idx=0; row_idx<submatrix_size; row_idx++) {
342  submatrix_prod[row_idx*submatrix_size+row_idx].real = submatrix_prod[row_idx*submatrix_size+row_idx].real - corner_element.real;
343  submatrix_prod[row_idx*submatrix_size+row_idx].imag = submatrix_prod[row_idx*submatrix_size+row_idx].imag - corner_element.imag;
344  }
345 
346  double unitary_error = cblas_dznrm2( submatrix_size*submatrix_size, submatrix_prod.get_data(), 1 );
347 
348  if (unitary_error < unitary_error_min) {
349  unitary_error_min = unitary_error;
350  memcpy(most_unitary_submatrix, submatrix, submatrix_size*submatrix_size*sizeof(QGD_Complex16));
351  }
352 
353  }
354  }
355 
356 
357  // if the qubit number in the submatirx is greater than 2 new N-qubit decomposition is started
358 
359  // create class tp decompose submatrices
360  N_Qubit_Decomposition* cdecomposition = new N_Qubit_Decomposition(most_unitary_submatrix_mtx, qbit_num-1, optimize_layer_num, initial_guess);
361 
362  // setting the verbosity
363  cdecomposition->set_verbose( verbose );
364 
365 
366  // Maximal number of iteartions in the optimization process
367  cdecomposition->set_max_iteration(max_iterations);
368 
369  // Set the number of identical successive blocks in the sub-decomposition
370  cdecomposition->set_identical_blocks(identical_blocks);
371 
372  // Set the maximal number of layers for the sub-decompositions
373  cdecomposition->set_max_layer_num(max_layer_num);
374 
375  // setting the iteration loops in each step of the optimization process
376  cdecomposition->set_iteration_loops( iteration_loops );
377 
378  // setting gate layer
379  cdecomposition->set_optimization_blocks( optimization_block );
380 
381  // set custom gate structure if given
382  cdecomposition->set_custom_gate_structure( gate_structure );
383 
384  // set the toleration of the optimization process
385  cdecomposition->set_optimization_tolerance( optimization_tolerance );
386 
387  // starting the decomposition of the random unitary
388  cdecomposition->start_decomposition(false, false);
389 
390 
391  // saving the decomposition gates
392  extract_subdecomposition_results( reinterpret_cast<Sub_Matrix_Decomposition*>(cdecomposition) );
393 
394  delete cdecomposition;
395

501 void N_Qubit_Decomposition::solve_layer_optimization_problem( int num_of_parameters, gsl_vector *solution_guess_gsl) {
502 
503  if (gates.size() == 0 ) {
504  return;
505  }
506 
507 
508  if (solution_guess_gsl == NULL) {
509  solution_guess_gsl = gsl_vector_alloc(num_of_parameters);
510  }
511 
512  if (optimized_parameters == NULL) {
513  optimized_parameters = (double*)qgd_calloc(num_of_parameters,sizeof(double), 64);
514  memcpy(optimized_parameters, solution_guess_gsl->data, num_of_parameters*sizeof(double) );
515  }
516 
517  // maximal number of iteration loops
518  int iteration_loops_max;
519  try {
520  iteration_loops_max = std::max(iteration_loops[qbit_num], 1);
521  }
522  catch (...) {
523  iteration_loops_max = 1;
524  }
525 
526  // do the optimization loops
527  for (int idx=0; idx<iteration_loops_max; idx++) {
528 
529  size_t iter = 0;
530  int status;
531 
532  const gsl_multimin_fdfminimizer_type *T;
533  gsl_multimin_fdfminimizer *s;
534 
535  N_Qubit_Decomposition* par = this;
536 
537 
538  gsl_multimin_function_fdf my_func;
539 
540 
541  my_func.n = num_of_parameters;
542  my_func.f = optimization_problem;
543  my_func.df = optimization_problem_grad;
544  my_func.fdf = optimization_problem_combined;
545  my_func.params = par;
546 
547 
548  T = gsl_multimin_fdfminimizer_vector_bfgs2;
549  s = gsl_multimin_fdfminimizer_alloc (T, num_of_parameters);
550 
551 
552  gsl_multimin_fdfminimizer_set (s, &my_func, solution_guess_gsl, 0.1, 0.1);
553 
554  do {
555  iter++;
556 
557  status = gsl_multimin_fdfminimizer_iterate (s);
558 
559  if (status) {
560  break;
561  }
562 
563  status = gsl_multimin_test_gradient (s->gradient, 1e-1);
564  /*if (status == GSL_SUCCESS) {
565  printf ("Minimum found\n");
566  }*/
567 
568  } while (status == GSL_CONTINUE && iter < 100);
569 
570  if (current_minimum > s->f) {
571  current_minimum = s->f;
572  memcpy( optimized_parameters, s->x->data, num_of_parameters*sizeof(double) );
573  gsl_multimin_fdfminimizer_free (s);
574 
575  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
576  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
577  }
578  }
579  else {
580  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
581  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
582  }
583  gsl_multimin_fdfminimizer_free (s);
584  }
585 
586 
587 
588  }

638 void
639 N_Qubit_Decomposition::simplify_layers() {
640 
641  if (verbose) {
642  printf("***************************************************************\n");
643  printf("Try to simplify layers\n");
644  printf("***************************************************************\n");
645  }
646 
647  // current starting index of the optimized parameters
648  int parameter_idx = 0;
649 
650  Gates_block* gates_loc = new Gates_block( qbit_num );
651  double* optimized_parameters_loc = (double*)qgd_calloc(parameter_num, sizeof(double), 64);
652  unsigned int parameter_num_loc = 0;
653 
654  unsigned int layer_idx = 0;
655 
656  while (layer_idx < gates.size()) {
657 
658  // generate a block of gates to be simplified
659  // (containg only successive two-qubit gates)
660  Gates_block* block_to_simplify = new Gates_block( qbit_num );
661 
662  std::vector<int> involved_qbits;
663  // layers in the block to be simplified
664  Gates_block* blocks_to_save = new Gates_block( qbit_num );
665 
666  // get the successive gates involving the same qubits
667  while (true) {
668 
669  if (layer_idx >= gates.size() ) {
670  break;
671  }

852  target_qbit = op->get_target_qbit();
853  control_qbit = op->get_control_qbit();

859  if ( (target_qbit == -1) || ( control_qbit == -1 ) ) {
860  return 1;
861  }
862

864 
865  // reorder the control and target qubits to the end of the list
866  std::vector<int> qbits_reordered;
867  for (int qbit_idx=qbit_num-1; qbit_idx>-1; qbit_idx-- ) { // in range(self.qbit_num-1,-1,-1):

893  int col_idx = element_idx % 4;
894  int row_idx = int((element_idx-col_idx)/4);
895  submatrix[element_idx].real = full_matrix_reordered[col_idx*matrix_size+row_idx].real;
896  submatrix[element_idx].imag = -full_matrix_reordered[col_idx*matrix_size+row_idx].imag;
897  }

917  printf("The simplification of the sub-structure was not possible\n");
