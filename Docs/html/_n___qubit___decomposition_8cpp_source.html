241 void N_Qubit_Decomposition::decompose_submatrix() {
242 
243  if (decomposition_finalized) {
244  if (verbose) {
245  printf("Decomposition was already finalized\n");
246  }
247  return;
248  }
249 
250  if (qbit_num == 2) {
251  return;
252  }
253 
254  // obtaining the subdecomposed submatrices
255  QGD_Complex16* subdecomposed_mtx = get_transformed_matrix( optimized_parameters, operations.begin(), operations.size(), Umtx );
256 
257  // get the most unitary submatrix
258  // get the number of 2qubit submatrices
259  int submatrices_num_row = 2;
260 
261  // get the size of the submatrix
262  int submatrix_size = int(matrix_size/2);
263 
264  // fill up the submatrices and select the most unitary submatrix
265 
266  QGD_Complex16* most_unitary_submatrix = (QGD_Complex16*)qgd_calloc( submatrix_size*submatrix_size,sizeof(QGD_Complex16), 64 );
267  double unitary_error_min = 1e8;
268 
269  for (int idx=0; idx<submatrices_num_row; idx++) { // in range(0,submatrices_num_row):
270  for (int jdx=0; jdx<submatrices_num_row; jdx++) { // in range(0,submatrices_num_row):
271 
272  QGD_Complex16* submatrix_prod = (QGD_Complex16*)qgd_calloc( submatrix_size*submatrix_size,sizeof(QGD_Complex16), 64 );
273  QGD_Complex16* submatrix = (QGD_Complex16*)qgd_calloc( submatrix_size*submatrix_size,sizeof(QGD_Complex16), 64 );
274 
275  for ( int row_idx=0; row_idx<submatrix_size; row_idx++ ) {
276  int matrix_offset = idx*(matrix_size*submatrix_size) + jdx*(submatrix_size) + row_idx*matrix_size;
277  int submatrix_offset = row_idx*submatrix_size;
278  memcpy(submatrix+submatrix_offset, subdecomposed_mtx+matrix_offset, submatrix_size*sizeof(QGD_Complex16));
279  }
280 
281  // calculate the product of submatrix*submatrix'
282  zgemm3m_wrapper_adj( submatrix, submatrix, submatrix_prod, submatrix_size);
283 
284  // subtract corner element
285  QGD_Complex16 corner_element = submatrix_prod[0];
286  for (int row_idx=0; row_idx<submatrix_size; row_idx++) {
287  submatrix_prod[row_idx*submatrix_size+row_idx].real = submatrix_prod[row_idx*submatrix_size+row_idx].real - corner_element.real;
288  submatrix_prod[row_idx*submatrix_size+row_idx].imag = submatrix_prod[row_idx*submatrix_size+row_idx].imag - corner_element.imag;
289  }
290 
291  double unitary_error = cblas_dznrm2( submatrix_size*submatrix_size, submatrix_prod, 1 );
292 
293  if (unitary_error < unitary_error_min) {
294  unitary_error_min = unitary_error;
295  memcpy(most_unitary_submatrix, submatrix, submatrix_size*submatrix_size*sizeof(QGD_Complex16));
296  }
297 
298  qgd_free(submatrix);
299  qgd_free(submatrix_prod);
300  submatrix = NULL;
301  submatrix_prod = NULL;
302  }
303  }
304 
305  qgd_free( subdecomposed_mtx );
306  subdecomposed_mtx = NULL;
307 
308 
309  // if the qubit number in the submatirx is greater than 2 new N-qubit decomposition is started
310 
311  // create class tp decompose submatrices
312  N_Qubit_Decomposition* cdecomposition = new N_Qubit_Decomposition(most_unitary_submatrix, qbit_num-1, optimize_layer_num, initial_guess);
313 
314  // setting the verbosity
315  cdecomposition->set_verbose( verbose );
316 
317 
318  // Maximal number of iteartions in the optimization process
319  cdecomposition->set_max_iteration(max_iterations);
320 
321  // Set the number of identical successive blocks in the sub-decomposition
322  cdecomposition->set_identical_blocks(identical_blocks);
323 
324  // Set the maximal number of layers for the sub-decompositions
325  cdecomposition->set_max_layer_num(max_layer_num);
326 
327  // setting the iteration loops in each step of the optimization process
328  cdecomposition->set_iteration_loops( iteration_loops );
329 
330  // setting operation layer
331  cdecomposition->set_optimization_blocks( optimization_block );
332 
333  // starting the decomposition of the random unitary
334  cdecomposition->start_decomposition(false, false);
335 
336 
337  // saving the decomposition operations
338  extract_subdecomposition_results( reinterpret_cast<Sub_Matrix_Decomposition*>(cdecomposition) );
339 
340  delete cdecomposition;
341  qgd_free( most_unitary_submatrix );
342  most_unitary_submatrix = NULL;

376 void N_Qubit_Decomposition::solve_layer_optimization_problem( int num_of_parameters, gsl_vector *solution_guess_gsl) {
377 
378  if (operations.size() == 0 ) {
379  return;
380  }
381 
382 
383  if (solution_guess_gsl == NULL) {
384  solution_guess_gsl = gsl_vector_alloc(num_of_parameters);
385  }
386 
387  if (optimized_parameters == NULL) {
388  optimized_parameters = (double*)qgd_calloc(num_of_parameters,sizeof(double), 64);
389  memcpy(optimized_parameters, solution_guess_gsl->data, num_of_parameters*sizeof(double) );
390  }
391 
392  // maximal number of iteration loops
393  int iteration_loops_max;
394  try {
395  iteration_loops_max = std::max(iteration_loops[qbit_num], 1);
396  }
397  catch (...) {
398  iteration_loops_max = 1;
399  }
400 
401  // do the optimization loops
402  for (int idx=0; idx<iteration_loops_max; idx++) {
403 
404  size_t iter = 0;
405  int status;
406 
407  const gsl_multimin_fdfminimizer_type *T;
408  gsl_multimin_fdfminimizer *s;
409 
410  N_Qubit_Decomposition* par = this;
411 
412 
413  gsl_multimin_function_fdf my_func;
414 
415 
416  my_func.n = num_of_parameters;
417  my_func.f = optimization_problem;
418  my_func.df = optimization_problem_grad;
419  my_func.fdf = optimization_problem_combined;
420  my_func.params = par;
421 
422 
423  T = gsl_multimin_fdfminimizer_vector_bfgs2;
424  s = gsl_multimin_fdfminimizer_alloc (T, num_of_parameters);
425 
426 
427  gsl_multimin_fdfminimizer_set (s, &my_func, solution_guess_gsl, 0.1, 0.1);
428 
429  do {
430  iter++;
431 
432  status = gsl_multimin_fdfminimizer_iterate (s);
433 
434  if (status) {
435  break;
436  }
437 
438  status = gsl_multimin_test_gradient (s->gradient, 1e-1);
439  /*if (status == GSL_SUCCESS) {
440  printf ("Minimum found\n");
441  }*/
442 
443  } while (status == GSL_CONTINUE && iter < 100);
444 
445 //printf("s->f: %f\n", s->f);
446  if (current_minimum > s->f) {
447  current_minimum = s->f;
448  memcpy( optimized_parameters, s->x->data, num_of_parameters*sizeof(double) );
449  gsl_multimin_fdfminimizer_free (s);
450 
451  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
452  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI/100;
453  }
454  }
455  else {
456  for ( int jdx=0; jdx<num_of_parameters; jdx++) {
457  solution_guess_gsl->data[jdx] = solution_guess_gsl->data[jdx] + (2*double(rand())/double(RAND_MAX)-1)*2*M_PI;
458  }
459  gsl_multimin_fdfminimizer_free (s);
460  }
461 
462 
463 
464  }

552 void N_Qubit_Decomposition::simplify_layers() {
553 
554  if (verbose) {
555  printf("***************************************************************\n");
556  printf("Try to simplify layers\n");
557  printf("***************************************************************\n");
558  }
559 
560  // current starting index of the optimized parameters
561  int parameter_idx = 0;
562 
563  Operation_block* operations_loc = new Operation_block( qbit_num );
564  double* optimized_parameters_loc = (double*)qgd_calloc(parameter_num, sizeof(double), 64);
565  unsigned int parameter_num_loc = 0;
566 
567  unsigned int layer_idx = 0;
568 
569  while (layer_idx < operations.size()) {
570 
571  // generate a block of operations to be simplified
572  // (containg only successive two-qubit operations)
573  Operation_block* block_to_simplify = new Operation_block( qbit_num );
574 
575  std::vector<int> involved_qbits;
576  // layers in the block to be simplified
577  Operation_block* blocks_to_save = new Operation_block( qbit_num );
578 
579  // get the successive operations involving the same qubits
580  while (true) {
581 
582  if (layer_idx >= operations.size() ) {
583  break;
584  }
585 
586  // get the current layer of operations
587  Operation* operation = operations[layer_idx];
588  layer_idx = layer_idx + 1;
589 
590  Operation_block* block_operation;
591  if (operation->get_type() == BLOCK_OPERATION) {
592  block_operation = static_cast<Operation_block*>(operation);
593  }
594  else {
595  layer_idx = layer_idx + 1;
596  continue;
597  }
598 
599 
600  //get the involved qubits
601  std::vector<int> involved_qbits_op = block_operation->get_involved_qubits();
602 
603  // add involved qubits of the operation to the list of involved qubits in the layer
604  for (std::vector<int>::iterator it=involved_qbits_op.begin(); it!=involved_qbits_op.end(); it++) {
605  add_unique_elelement( involved_qbits, *it );
606  }
607 
608  if ( (involved_qbits.size())> 2 && blocks_to_save->get_operation_num() > 0 ) {
609  layer_idx = layer_idx -1;
610  involved_qbits.clear();
611  break;
612  }
613 
614  blocks_to_save->combine(block_operation);
615 
616  // adding the operations to teh block if they act on the same qubits
617  block_to_simplify->combine(block_operation);
618 
619 
620  }
621 
622  //number of perations in the block
623  unsigned int parameter_num_block = block_to_simplify->get_parameter_num();
624 
625  // get the number of CNOT gates and store the block operations if the number of CNOT gates cannot be reduced
626  gates_num gate_nums = block_to_simplify->get_gate_nums();
627 
628  if (gate_nums.cnot < 2) {
629  operations_loc->combine(blocks_to_save);
630  memcpy(optimized_parameters_loc+parameter_num_loc, optimized_parameters+parameter_idx, parameter_num_block*sizeof(double) );
631  parameter_idx = parameter_idx + parameter_num_block;
632  parameter_num_loc = parameter_num_loc + parameter_num_block;
633 
634  if ( block_to_simplify != NULL ) {
635  delete block_to_simplify;
636  block_to_simplify = NULL;
637  }
638 
639  if ( blocks_to_save != NULL ) {
640  delete blocks_to_save;
641  blocks_to_save = NULL;
642  }
643 
644  continue;
645  }
646 
647 
648  // simplify the given layer
649  std::map<int,int> max_layer_num_loc;
650  max_layer_num_loc.insert( std::pair<int, int>(2, gate_nums.cnot-1 ) );
651  Operation_block* simplified_layer = NULL;
652  double* simplified_parameters = NULL;
653  unsigned int simplified_parameter_num=0;
654 
655  // Try to simplify the sequence of 2-qubit operations
656  int simplification_status = simplify_layer( block_to_simplify, optimized_parameters+parameter_idx, parameter_num_block, max_layer_num_loc, simplified_layer, simplified_parameters, simplified_parameter_num );
657 //simplification_status = -1;
658 
659 
660  // adding the simplified operations (or the non-simplified if the simplification was not successfull)
661  if (simplification_status == 0) {
662  operations_loc->combine( simplified_layer );
663 
664  if (parameter_num < parameter_num_loc + simplified_parameter_num ) {
665  optimized_parameters_loc = (double*)qgd_realloc( optimized_parameters_loc, parameter_num, parameter_num_loc + simplified_parameter_num, sizeof(double), 64 );
666  }
667  memcpy(optimized_parameters_loc+parameter_num_loc, simplified_parameters, simplified_parameter_num*sizeof(double) );
668  parameter_num_loc = parameter_num_loc + simplified_parameter_num;
669  }
670  else {
671  // addign the stacked operation to the list, sice the simplification was unsuccessful
672  operations_loc->combine( blocks_to_save );
673 
674  if (parameter_num < parameter_num_loc + parameter_num_block ) {
675  optimized_parameters_loc = (double*)qgd_realloc( optimized_parameters_loc, parameter_num, parameter_num_loc + parameter_num_block, sizeof(double), 64 );
